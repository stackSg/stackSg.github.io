递归

定义

一种直接或间接调用自身的函数称为递归函数，使用递归可以解决每次面临同一问题，由较简单到最简单的情况，需要注意的是递归的边界条件，也就是终止条件。

思考步骤

  1. 找重复–子问题（在总问题中划一刀，切出子问题）
  2. 找变化：变化的量作为参数（递归写不出来的时候思考是否缺了变化参数）
  3. 找边界：出口
    重复中的变化，变化中的重复
    分解为：直接量+小规模子问题
    分解为：多个小规模子问题
    递归树：先纵后横

递归算法的时间复杂度分析

示例

  1. 斐波那契数列

    • 指的是这样一个数列：1、1、2、3、5、8、13、21、34、……在数学上，斐波那契数列以如下被以递推的方法定义：F(0)=0，F(1)=1, F(n)=F(n-1)+F(n-2)（n>=3，n∈N*）

    • 迭代方式代码实现

      int main()
{
  int a = 0;    //定义初始值 
  int b = 1;
  int c;

  printf("%d\t%d\t", a, b);        //打印数列第0，1项 
  for (int i = 2; i <= 40; i++)    //打印数列前40项 
  {
      c = a + b;                    //后一项数值等于前两项之和 
      printf("%d\t", c);
      a = b;                        //用a变量存储第n-2项 
      b = c;                        //用b变量存储第n-1项 
  }

  printf("\n");
}

    • 递归方式代码实现

      int fib(int i)
{
  if (i < 2)                        //递归的终止条件为第数列的前两项 
  {
      return i == 0 ? 0 : 1;
  }

  return fib(i - 1) + fib(i - 2);    // F(n)=F(n-1)+F(n-2)
}

    • 时间复杂度：T(n) = 2T(n - 1) + O(1) –> O(2^n)

  2. 逆序打印输入的字符串

    • 若输入”abc”则打印出”cba”，以’#’作为结束字符。

    • 递归实现

      void print()
{
  char c;
  scanf("%c", &c);

  if (c != '#')    //如果输入字符不是结束标识则继续输入 
  {
      print();
  }
  if (c != '#')    //当输入结束标识后递归返回到此处打印输入的字符 
  {
      printf("%c", c);
  }
}

  3. 最大公约数

    • 递推公式：f(m,n)=f(n,m%n)
    • 递归实现：
        int gcc(int m, int n)
  {
      if (n == 0)
      {
          return m;
      }
      return gcc(n, m % n);
  }
    • 时间复杂度：O(2logn)

  4. 插入排序

    • 对k-1个元素进行插入排序，将第k个元素插入k-1个有序元素中
    • 递归实现
      static void insertSort(int[] arr, int k) {
  if (k == 0) {
      return;
  }
  //对k-1个元素排序
  insertSort(arr, k - 1);
  // 把第k个元素插入到前面k-1个有序序列中
  int index = k - 1;
  int x = arr[k];
  while (index >= 0 && x < arr[index]) {
      arr[index + 1] = arr[index];
      index--;
  }
  arr[index + 1] = x;
}

分治思想

二分查找

  • 二分查找也称折半查找（Binary Search），它是一种效率较高的查找方法。但是，折半查找要求线性表必须采用顺序存储结构，而且表中元素按关键字有序排列。

  • 迭代方式

    int BinarySearch(int a[], int length, int num)
{
  int low = 0;
  int high = length - 1;    //初始值low和high指向数组头和尾 
  int mid;                //存储中间位置的下标 

  while (low <= high)
  {
      mid = (low + high) / 2;        //计算出折半的位置 
      if (a[mid] > num)            //如果中间值大于指定值则查找空间锁定到low-(mid-1)
      {
          high = mid - 1;
      }
      if (a[mid] < num)            //如果中间值小于指定值则查找空间锁定到(mid+1)-high 
      {
          low = mid + 1;
      }
      if (a[mid] == num)            //如果查找到指定值则返回下标 
      {
          return mid;
      }
  }    

  return -1;
}

  • 递归方式

    int BinarySearch(int a[], int low, int high, int num)
{
  if (low > high)                //递归终止条件为未找到或找到返回 
  {
      return -1;
  }

  int mid = (low + high) / 2;    //计算折半下标 
  if (a[mid] == num)
  {
      return mid;
  }
  if (a[mid] > num)
  {
      return BinarySearch(a, low, mid - 1, num);
  }
  if (a[mid] < num)
  {
      return BinarySearch(a, mid + 1, high, num);
  }
}

汉诺塔

  • 汉诺塔：汉诺塔（又称河内塔）问题是源于印度一个古老传说的益智玩具。大梵天创造世界的时候做了三根金刚石柱子，在一根柱子上从下往上按照大小顺序摞着64片黄金圆盘。大梵天命令婆罗门把圆盘从下面开始按大小顺序重新摆放在另一根柱子上。并且规定，在小圆盘上不能放大圆盘，在三根柱子之间一次只能移动一个圆盘。

  • 递归实现

      #include <stdio.h>

  /**
  *    n:汉诺塔的层数
  *    x,y,z:三根柱子 
  *    在这里指定从x移动到z 
  **/
  void move(int n, char x, char y, char z)
  {
      if (n == 1)    //递归终止条件为只有一个盘子x--->z 
      {
          printf("%c--->%c\n", x, z);
      }
      else
      {
          move(n - 1, x, z, y);        //先将第n个盘子上面的n-1个盘子x--->y 
          printf("%c--->%c\n", x, z);    //将第n个盘子x--->z 
          move(n - 1, y, x, z);        //将y上的n-1个盘子y--->z 
      }
  }

  int main()
  {
      int n;
      printf("请输入汉诺塔的层数：");
      scanf("%d", &n);
      move(n, 'X', 'Y', 'Z');
  }

八皇后问题

  • 八皇后问题，是一个古老而著名的问题，是回溯算法的典型案例。该问题是国际西洋棋棋手马克斯·贝瑟尔于1848年提出：在8×8格的国际象棋上摆放八个皇后，使其不能互相攻击，即任意两个皇后都不能处于同一行、同一列或同一斜线上，问有多少种摆法。 高斯认为有76种方案。1854年在柏林的象棋杂志上不同的作者发表了40种不同的解，后来有人用图论的方法解出92种结果。计算机发明后，有多种计算机语言可以解决此问题。

  • 递归实现

      #include <stdio.h>

  /**
      row:行坐标
      n:列坐标
      chess:棋盘 
  **/ 
  int IsSafe(int row, int n, int chess[][8])
  {
      int i, j;
      //判断列方向是否安全 
      for (i = 0, j = n; i < 8; i++)
      {
          if (chess[i][j] != 0)
          {
              return 0;
          }
      }

      //判断左上方
      for (i = row, j = n; i >= 0 && j >= 0; i--, j--)
      {
          if (chess[i][j] != 0)
          {
              return 0;
          }
      } 

      //判断右下方 
      for (i = row, j = n; i < 8 && j < 8; i++, j++)
      {
          if (chess[i][j] != 0)
          {
              return 0;
          }
      } 

      //判断左下方
      for (i = row, j = n; i < 8 && j >= 0; i++, j--)
      {
          if (chess[i][j] != 0)
          {
              return 0;
          }
      } 

      //判断右上方
      for (i = row, j = n; i >= 0 && j < 8; i--, j++)
      {
          if (chess[i][j] != 0)
          {
              return 0;
          }
      } 

      return 1;
  }

  int count = 0;    //记录总共有多少种解法 

  /*
      row : 当前行坐标
      n   : 当前列数
      chess:  棋盘 
  */
  void EightQueens(int row, int n, int chess[][8])
  {
      int chess2[8][8];
      for (int i = 0; i < 8; i++)
      {
          for (int j = 0; j < 8; j++)
          {
              chess2[i][j] = chess[i][j];
          }
      }

      if (row == 8)
      {
          printf("第%d种解法:\n", count + 1);
          for (int i = 0; i < 8; i++)
          {
              for (int j = 0; j < 8; j++)
              {
                  printf("%d ", chess2[i][j]);
              }
              printf("\n");
          }
          printf("\n");

          count++;
      }
      else
      {
          for (int i = 0; i < n; i++)
          {
              if (IsSafe(row, i, chess2))
              {
                  for (int j = 0; j < 8; j++)
                  {
                      chess2[row][j] = 0;
                  }
                  chess2[row][i] = 1;
                  EightQueens(row + 1, n, chess2);
              }
          }
      }
  }

  int main()
  {
      int chess[8][8] = {0};

      EightQueens(0, 8, chess);
  }
