散列表

  • 散列表（Hash table，也叫哈希表），是根据关键码值(Key value)而直接进行访问的数据结构。也就是说，它通过把关键码值映射到表中一个位置来访问记录，以加快查找的速度。这个映射函数叫做散列函数，存放记录的数组叫做散列表。

散列表的两个关键点

  • 散列函数的构造方法
  • 处理散列冲突的方法

散列函数的构造

  • 构造散列函数的两个基本原则：
    • 计算简单
    • 分布均匀
  • 根据不同情况采用不同的散列函数：
    • 计算散列地址所需要的时间
    • 关键字的长度
    • 散列表的大小
    • 关键字的分布情况
    • 记录查找的频率

示例代码

#include <stdio.h>
#include <stdlib.h>

#define HASHSIZE 12        //定义散列表空间大小 
#define NULLKEY -32768    //散列表元素初始值 

typedef struct 
{
    int *elem;    //散列表空间基址 
    int count;    //散列表总长度 
} HashTable;

//散列表初始化 
int InitHashTable(HashTable *hash)
{
    hash->elem = (int *)malloc(sizeof(int) * HASHSIZE);
    if (!hash->elem)
    {
        return -1;    //申请空间失败，返回错误代码 
    }

    for (int i = 0; i < HASHSIZE; i++)
    {
        hash->elem[i] = NULLKEY;
    } 

    hash->count = HASHSIZE;

    return 0; 
} 

//除留余数法 
int Hash(int key)
{
    return key % (HASHSIZE - 1);    //余数大小一般<=散列表长度且为质数 
} 

//散列表插入 
int InsertHash(HashTable *hash, int key)
{
    int addr = Hash(key);

    while (hash->elem[addr] != NULLKEY)        
    {
        addr = (addr + 1) % (HASHSIZE - 1);    //冲突解决使用线性探测法 
        if (addr == Hash(key))                // 偏移地址重新回到起始地址证明散列表已满 
        {
            return -1;
        }
    }

    hash->elem[addr] = key;

    return 0;
}

//散列表查找 
int SearchHash(HashTable *hash, int key, int *addr)
{
    *addr = Hash(key);

    while (hash->elem[*addr] != key)    //如果偏移地址定位的元素不等于key值则采用线性探测法继续查找 
    {
        *addr = (*addr + 1) % (HASHSIZE - 1);
        if (hash->elem[*addr] == NULLKEY || *addr == Hash(key))     //查找到初始值的情况或回到首次hash的位置都为查找失败的情况 
        {
            return -1;
        }
    }

    return 0;
}

void PrintHash(HashTable *hash)
{
    for (int i = 0; i < hash->count; i++)
    {
        if (hash->elem[i] != NULLKEY)
        {
            printf("HashTable[%d] = %d\n", i, hash->elem[i]);
        }
        else
        {
            printf("HashTable[%d] = NULL\n", i);
        }
    }
}

int main()
{
    HashTable *hash = (HashTable *)malloc(sizeof(HashTable));

    if (InitHashTable(hash))
    {
        return -1;
    }

    int keys[] = { 5, 11, 10, 18, 11, 6, };
    int key = 10;

    for(int i = 0; i < 6; i++)
    {
        if (InsertHash(hash, keys[i]))
        {
            printf("散列表已满\n");
            break;
        }
    }

    PrintHash(hash);

    int addr;
    int result = SearchHash(hash, key, &addr);

    if (result)
    {
        printf("散列表中不存在key值:%d\n", key);
        return -1;
    }

    printf("查找到的key=%d\n", hash->elem[addr]);
}
