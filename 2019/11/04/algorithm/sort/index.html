常见排序算法的分类

基本概念

  • 稳定排序：假设在待排序的文件中，存在两个或两个以上的记录具有相同的关键字，在用某种排序法排序后，若这些相同关键字的元素的相对次序仍然不变，则这种排序方法是稳定的。

衡量排序算法优劣的性能指标

  • 时间复杂度
  • 空间复杂度
  • 稳定性

有关时间复杂度和空间复杂度的概念参考这篇文章：https://blog.csdn.net/jsjwk/article/details/84315770

交换排序

  • 冒泡排序

      void BubbleSort(int a[], int n)
  {
      int t;

      for (int i = 0; i < n - 1; i++)
      {
          for (int j = 0; j < n - i; j++)
          {
              if (a[j] > a[j + 1])
              {
                  t = a[j];
                  a[j] = a[j + 1];
                  a[j + 1] = t;
              }
          }
      }
  }

  • 快速排序

      void QuickSort(int a[], int left, int right)
  {
      if (left >= right)
      {
          return;
      }

      int i = left;
      int j = right;
      int key = a[left];

      while (i < j)
      {
          while (i < j && a[j] >= key)
          {
              j--;
          }
          a[i] = a[j];

          while (i < j && a[i] <= key)
          {
              i++;
          }
          a[j] = a[i];
      }
      a[i] = key;

      QuickSort(a, left, i - 1);
      QuickSort(a, i + 1, right);
  }

  • 鸡尾酒排序

      void CocktailSort(int a[], int n)
  {
      int left = 0;
      int right = n - 1;
      int t;

      while (left < right)                    
      {
          for (int i = left; i < right; i++)
          {
              if (a[i] > a[i + 1])
              {
                  t = a[i];
                  a[i] = a[i + 1];
                  a[i + 1] = t;
              }
          }
          right--;

          for (int j = right; j > left; j--)
          {
              if (a[j] < a[j - 1])
              {
                  t = a[j];
                  a[j] = a[j - 1];
                  a[j - 1] = t;
              }
          }
          left++;
      }
  }

选择排序

  • 简单选择排序

      void SelectionSort(int a[], int n)
  {
      int t;
      int minind;

      for (int i = 0; i < n - 1; i++)
      {
          minind = i;
          for (int j = i + 1; j < n; j++)
          {
              if (a[j] < a[minind])
              {
                  minind = j;
              }
          }
          if (minind != i)
          {
              t = a[minind];
              a[minind] = a[i];
              a[i] = t;
          }
      }
  }

插入排序

  • 直接插入排序

      void InsertSort(int a[], int n)
  {
      int t;
      int min;

      for (int i = 1, j; i < n; i++)
      {
          t = a[i];
          j = i - 1;
          while(j >= 0 && t < a[j])
          {
              a[j + 1] = a[j];
              j--;
          }
          a[j + 1] = t;
      }
  }

  • 希尔排序

      void ShellSort(int a[], int n)
  {
      int k, t;

      for (int i = n / 2; i > 0; i /= 2)
      {
          for (int j = i; j < n; j++)
          {
              t = a[j];
              k = j - i;
              while (k >= 0 && a[k] > t)
              {
                  a[k + i] = a[k];
                  k -= i;
              }
              a[k + i] = t;
          }
      }
  }

基数排序

void radixSort(int a[], int n)
{
    int max = MaxArray(a, n);

    int maxloop = NumLoop(max);

    for (int i = 1; i <= maxloop; i++)
    {
        sort(a, n, i);
    }
}
int MaxArray(int a[], int n)
{
    int maxind = 0;

    for (int i = 1; i < n; i++)
    {
        if (a[maxind] < a[i])
        {
            maxind = i;
        }
    }

    return a[maxind];
}

int NumLoop(int n)
{
    int count = 0;

    do
    {
        n /= 10;
        count++;
    }while (n != 0);

    return count;
}

void sort(int a[], int n, int x)
{
    int b[10][5] = {0};
    int j;

    int p = (int)pow(10, x - 1);
    int row_index;

    for (int i = 0; i < n; i++)
    {
        row_index = (a[i] / p) % 10;
        for (j = 0; j < 5; j++)
        {
            if (b[row_index][j] == 0)
            {
                b[row_index][j] = a[i];
                break;
            }
        }
    }

    int k = 0;
    for (i = 0; i < 10; i++)
    {
        for (j = 0; j < 5; j++)
        {
            if (b[i][j] != 0)
            {
                a[k++] = b[i][j];
            }
        }
    }
}

鸽巢排序

void Pigeonholesort(int a[], int n)
{
    int b[100] = {0};

    for (int i = 0; i < n; i++)
    {
        b[a[i]]++;
    }

    i = 0;
    for (int j = 0; j < 100; j++)
    {
        for (int k = 0; k < b[j]; k++)        //在此用循环是考虑到相等值的情况
        {
            a[i++] = j;
        }
    }
}
