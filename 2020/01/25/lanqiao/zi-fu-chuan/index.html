  1. 替换字符串中的空格

    • 给定字符串中的空格替换为%20

    • 解：

      //4.将给定字符串中的空格替换为%20
static String replaceSpace(char[] s, int length){
  //计算替换后字符串长度
  int count = length;
  for (int i = 0; i < length; i++){
      if (s[i] == ' '){
          count += 2;
      }
  }

  //原始字符串末尾
  int p1 = length - 1;
  //新增空间末尾
  int p2 = count - 1;

  //将原始字符串从末尾依次复制到新增空间，遇到空格替换为%20
  while (p1 >= 0){
      if (s[p1] == ' '){
          s[p2--] = '0';
          s[p2--] = '2';
          s[p2--] = '%';
      }else{
          s[p2--] = s[p1];
      }
      p1--;
  }

  return new String(s, 0, count);
}

      //5.压缩字符串
static String zipString(String s){
  StringBuilder sb = new StringBuilder();
  int count = 0;
  char last = ' ';

  for (int i= 0; i < s.length(); i++){
      char c = s.charAt(i);

      //如果是第一个字符则直接加入
      if (sb.length() == 0){
          sb.append(c);
          count++;
      }else{
          //如果上一个字符和本字符相同则累加
          if (last == c){
              count++;
          }else{
              //否则将上一字符的出现次数写入并重新开始统计本字符
              sb.append(count);
              sb.append(c);
              count = 1;
          }
      }

      last = c;
  }
  //将最后一个字符的出现次数写入
  sb.append(count);

  if (sb.length() >= s.length()){
      return s;
  }else{
      return sb.toString();
  }
}

static String removeZero2(String s, int k){

      if (arr[i] == '0'){

  7. 最短摘要的生成

    • Alibaba笔试题：给定一段产品的英文描述，包含M个英文字母，每个英文单词以空格分隔，无其他标点符号；再给定N个英文单词关键字，请说明思路并编程实现方法
      String extractSummary(String description,String[] key words)
      目标是找出此产品描述中包含N个关键字（每个关键词至少出现一次）的长度最短的子串，作为产品简介输出。（不限编程语言）20分。

      public class 最短摘要 {
  public static void main(String[] args) {
      String s = "a b c speed h e f c c seed e f c";
      String[] key = {
              "h",
              "speed",
              "c",
      };
      System.out.println(extractSummary(s, key));
      System.out.println(solve(s, key));
  }
  //暴力枚举
  static String solve(String description,String[] keywords){
      String[] desc = description.split(" ");
      int start = -1;
      int end = -1;
      //枚举以i位置开始的字符串
      for (int i = 0; i < desc.length; i++) {
          //向后查找包含所有关键字的j位置
          for (int j = i + 1; j < desc.length; j++) {
              if (containsAll(desc, keywords, i, j)){
                  //更新为最短字符串
                  if (start == -1 || end - start > j - i){
                      start = i;
                      end = j;
                  }
              }
          }
      }

      StringBuilder res = new StringBuilder();
      for (int k = start; k <= end; k++) {
          res.append(desc[k] + " ");
      }
      res.deleteCharAt(res.length() - 1);

      return res.toString();
  }
  //尺取法
  static String extractSummary(String description,String[] keywords){
      String[] desc = description.split(" ");
      int i = 0;
      int j = -1;
      int begin = -1;
      int end = -1;
      //将关键字存储为hash映射
      Set<String> keys = new HashSet<>();
      for (int k = 0; k < keywords.length; k++) {
          keys.add(keywords[k]);
      }

      while (i < desc.length && j < desc.length){
          //i在文章中找到第一个是关键字的位置
          while (i < desc.length && !keys.contains(desc[i])){
              i++;
          }
          //j的起始位置为i的后面
          if (j == -1){
              j = i + 1;
          }
          //j向后找到包含全部关键字的位置
          while (j < desc.length && !containsAll(desc, keywords, i, j)){
              j++;
          }
          //如果包含所有关键字的位置存在
          if (i < desc.length && j < desc.length){
              //更新为最短的位置
              if (begin == -1 || end - begin > j - i){
                  begin = i;
                  end = j;
              }
          }
          //i向后继续寻找更短的位置
          i++;
      }
      //还原最短摘要
      StringBuilder res = new StringBuilder();
      for (int k = begin; k <= end; k++) {
          res.append(desc[k] + " ");
      }
      res.deleteCharAt(res.length() - 1);

      return res.toString();
  }
  //判断desc数组中start-end下标之间是否包含keys中的全部内容
  static boolean containsAll(String[] desc, String[] keys, int start, int end){
      //标识关键字是否出现
      boolean[] f = new boolean[keys.length];
      //枚举每一个关键字
      for (int i = 0; i < keys.length; i++) {
          //在范围内查找该关键字
          for (int j = start; j <= end; j++) {
              //如果关键字出现则标识
              if (desc[j].equals(keys[i])){
                  f[i] = true;
              }
          }
      }
      //判断所有关键字是否出现
      for (int i = 0; i < f.length; i++) {
          if (!f[i]){
              return false;
          }
      }
      return true;
  }
}

后缀数组

  • 将源字符串的所有后缀组合为后缀数组，对后缀数组进行字典排序，使用二分查找法在后缀数组中匹配模式串。

  • 模式串一定为某后缀串的前缀

  • 实现：

    class Suff implements Comparable<Suff> {
  String str;    //后缀子串
  int index;    //原始下标

  public Suff(String str, int index) {
      this.str = str;
      this.index = index;
  }

  @Override
  public int compareTo(Suff o) {
      return this.str.compareTo(o.str);
  }

  @Override
  public String toString() {
      return "Suff [str=" + str + ", index=" + index + "]";
  }
}
//得到后缀数组
static Suff[] getSa(String s){
  Suff[] arr = new Suff[s.length()];

  for (int i = 0; i < s.length(); i++) {
      arr[i] = new Suff(s.substring(i), i);
  }
  //对后缀数组字典排序
  Arrays.sort(arr);

  return arr;
}
//后缀数组查找子串，采用二分查找法
static int suffixArray(String p, String s){
  int i = 0;
  int j = s.length() - 1;

  Suff[] arr = getSa(s);
  int result;
  while (i <= j){
      int mid = (i + j) / 2;
      //子串一定是后缀串的前缀，所以长度不相等的情况下截取前缀比较
      if (p.length() < arr[mid].str.length()){
          result = arr[mid].str.substring(0, p.length()).compareTo(p);
      }else{
          result = arr[mid].str.compareTo(p);
      }

      if (result == 0){
          return arr[mid].index;
      }else if (result > 0){
          j = mid - 1;
      }else{
          i = mid + 1;
      }
  }

  return -1;
}

  • Arrary.sort()使用快速排序，所以复杂度为O(nlong(n))，但是因为字符串比较大小是采用逐位比较的方法，复杂度为O(n)，综合下来以上算法实现的复杂度为：O(N^2*log(N))。

  • 优化

    • 从后缀数组排序入手降低复杂度

    • 使用倍增法求后缀数组

      // 倍增法
static suff[] suffixArray2(String s) {
  suff arr[] = new suff[s.length()];
  int[] rank = new int[s.length()];

  // 倍增量为1的时候取每个位置的字符
  for (int j = 0; j < s.length(); j++) {
      arr[j] = new suff(s, j, s.charAt(j));
  }

  // 单个字符直接排序
  Arrays.sort(arr, new Comparator<suff>() {

      @Override
      public int compare(suff o1, suff o2) {
          // TODO Auto-generated method stub
          return o1.c - o2.c;
      }
  });

  // 得到初始的rank数组
  rank[arr[0].index] = 1;
  for (int j = 1; j < arr.length; j++) {
      if (arr[j - 1].c == arr[j].c) {
          rank[arr[j].index] = rank[arr[j - 1].index];
      } else {
          rank[arr[j].index] = rank[arr[j - 1].index] + 1;
      }
  }

  for (int i = 2; i <= s.length(); i *= 2) {
      final int ii = i;
      // 根据rank数组排序
      Arrays.sort(arr, new Comparator<suff>() {
          @Override
          public int compare(suff o1, suff o2) {
              int m = o1.index;
              int n = o2.index;
              // 比较倍增之前的排名
              if (rank[m] == rank[n]) {
                  // 如果相同根据增加后的字符比较大小
                  // 如果没有第二关键字则一定是它最小 索引靠后
                  if (m + ii / 2 >= rank.length || n + ii / 2 >= rank.length) {
                      return -(m - n);
                  } else {
                      return rank[m + ii / 2] - rank[n + ii / 2];
                  }
              } else {
                  // 如果不同根据之前排名直接比较大小
                  return rank[m] - rank[n];
              }
          }
      });
      // 得到rank数组
      rank[arr[0].index] = 1;
      for (int j = 1; j < arr.length; j++) {
          rank[arr[j].index] = rank[arr[j - 1].index];
          try{
              //因为suf只保存了后缀串的第一个字符 所以在比较两个后缀串是否相同时需要使用截取时的下标和增量重新截取比较
              if (!s.substring(arr[j - 1].index, arr[j - 1].index + i).equals(s.substring(arr[j].index, arr[j].index + i))) {
                  rank[arr[j].index] = rank[arr[j - 1].index] + 1;
              }
          }catch (StringIndexOutOfBoundsException e){
              rank[arr[j].index] = rank[arr[j - 1].index] + 1;
          }
      }
  }

  return arr;
}

    • 求高度数组

      //求后缀数组的高度数组
static int[] getHight(String s, suff[] suf){
  //得到后缀数组的排名
  int[] rank = new int[s.length()];
  for (int i = 0; i < rank.length; i++) {
      rank[suf[i].index] = i;
  }

  int[] height = new int[s.length()];
  //k为上一排名的LCP值
  int k = 0;
  for (int i = 0; i < suf.length; i++) {
      //得到i起点为后缀的排名
      int rk_i = rank[i];
      //如果排名为1则它的LCP=0
      if (rk_i == 0){
          height[rk_i] = 0;
          continue;
      }
      //得到上一排名的后缀串
      int rk_i_1 = rk_i - 1;
      //得到上一排名后缀串的起点
      int j = suf[rk_i_1].index;
      //LCP(i, i-1) >= LCP(i-1, i-2) - 1
      if (k > 0){
          k--;
      }
      //从上一个LCP值相加为起始点去求得本次LCP值
      while (i + k < s.length() && j + k < s.length()){
          if (s.charAt(i + k) != s.charAt(j + k)){
              break;
          }
          k++;
      }
      //赋值LCP值
      height[rk_i] = k;
  }

  return height;
}

  1. hiho字符串

    • 原题：hihocoder hiho字符串

    • 如果一个字符串恰好包含2个’h’、1个’i’和1个’o’，我们就称这个字符串是hiho字符串。例如”oihateher”、”hugeinputhugeoutput”都是hiho字符串。现在给定一个只包含小写字母的字符串S，小Hi想知道S的所有子串中，最短的hiho字符串是哪个。

    • 解：

      public static void main(String[] args) {
  Scanner input = new Scanner(System.in);
  String s = input.next();
  System.out.println(hiho(s));
}

// hiho字符串
static int hiho(String s) {
  char chars[] = s.toCharArray();

  int j = -1;
  int min = Integer.MAX_VALUE;
  for (int i = 0; i < chars.length; i++) {
      //i找到包含字符后固定
      if (check(chars[i])) {
          if (j == -1) {
              j = i + 1;
          }
          //j向前移动，直到包含关键字符
          while (j < chars.length) {
              if (check(chars[j]) && containsAll(chars, i,j)) {
                  //判断是否恰好包含关键字符并且为最短长度
                  int length = j - i + 1;
                  if (check(chars, i ,j) && length < min) {
                      min = length;
                  }
                  //找到包含关键字符的一截字符串后，j固定，i继续向前移动缩小范围
                  break;
              }
              j++;
          }
      }
  }

  if (min == Integer.MAX_VALUE) {
      return -1;
  } else {
      return min;
  }
}

//判断是否恰好包含关键字符
private static boolean check(char[] chars, int x, int y) {
  int a = 0, b = 0, c = 0;
  for (int i = x; i <= y; i++) {
      if (chars[i] == 'h')
          a++;
      if (chars[i] == 'i')
          b++;
      if (chars[i] == 'o')
          c++;
  }
  return a == 2 && b == 1 && c == 1;
}

//判断是否包含关键字符
private static boolean containsAll(char[] chars, int x, int y) {
  int a = 0, b = 0, c = 0;
  for (int i = x; i <= y; i++) {
      if (chars[i] == 'h')
          a++;
      if (chars[i] == 'i')
          b++;
      if (chars[i] == 'o')
          c++;
  }
  return a >= 2 && b >= 1 && c >= 1;
}

//判断是否为关键字符
static boolean check(char c) {
  return c == 'h' || c == 'i' || c == 'o';
}

  1. 前缀周期性

    • 原题：vjudge 杭电-1358

    • 解：

      // HDU-1358 前缀的重复周期
static void perfix() {
  //接收输入字符串
  Scanner input = new Scanner(System.in);
  int n = input.nextInt();
  ArrayList<String> list = new ArrayList<>();

  while (n != 0){
      String s = input.next();
      //存入list
      list.add(s);
      n = input.nextInt();
  }

  int count = 1;
  //遍历输入字符串，分别求得前缀周期
  for (String str : list) {
      //得到字符串的next数组
      int[] next = getNext(str);

      System.out.println("Test case #" + count++);
      //遍历next数组
      for (int i = 0; i < next.length; i++) {
          //t为重复串的长度
          int t = i - next[i];
          //判断整个字符串的长度是否为单个重复串的整数倍且倍数为2及以上
          if (i % t == 0 && i / t > 1){
              //打印当前长度和重复次数
              System.out.println(i + " " + (i / t));
          }
      }
  }
}

private static int[] getNext(String str) {
  char[] chars = str.toCharArray();
  int[] next = new int[str.length() + 1];
  next[0] = -1;

  if (str.length() == 1){
      return next;
  }

  next[1] = 0;
  int j = 1;
  int k = next[j];

  while (j + 1 < next.length){
      if (k < 0 || chars[j] == chars[k]){
          next[++j] = ++k;
      }else{
          k = next[k];
      }
  }

  return next;
}
