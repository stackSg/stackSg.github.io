顺序存储结构实现

/**
 * 顺序存储方式的列表(动态数组)
 * @author 96047
 *
 * @param <T>
 */
public class MyArrayList<T> implements MyList<T> {
    //列表
    private T[] list;
    //元素个数
    private int size;
    //容量
    private int capacity = 10;

    //默认初始化
    public MyArrayList(){
        list = (T[]) new Object[capacity];
    }

    //根据传入容量初始化
    public MyArrayList(int capacity){
        this.capacity = capacity;
        list = (T[]) new Object[capacity];
    }

    @Override
    public void add(T element) {
        //如果容量没有了则扩容
        if (size == capacity){
            capacity *= 2;
            T[] cList = (T[]) new Object[capacity];
            for (int i = 0; i < size; i++) {
                cList[i] = list[i];
            }
            list = cList;
        }
        list[size++] = element;
    }

    @Override
    public void delete(T element) {
        int index = indexOf(element);
        if (index >= 0){
            delete(index);
        }
    }

    @Override
    public void delete(int index) {
        for (int i = index; i < size - 1; i++) {
            list[i] = list[i + 1];
        }
        list[size - 1] = null;
        size--;
    }

    @Override
    public void update(int index, T element) {
        list[index] = element;
    }

    @Override
    public boolean contains(T element) {
        return indexOf(element) >= 0;
    }

    @Override
    public T at(int index) {
        return list[index];
    }

    @Override
    public int indexOf(T element) {
        for (int i = 0; i < size; i++) {
            if (list[i].equals(element)){
                return i;
            }
        }
        return -1;
    }

    @Override
    public String toString() {
        StringBuilder res = new StringBuilder("[");
        for (int i = 0; i < size; i++) {
            res.append(list[i] + ",");
        }
        res.deleteCharAt(res.length() - 1);
        res.append("]");

        return res.toString();
    }

    @Override
    public int length() {
        return size;
    }
}

/**
 * 单链表(无头)
 * @author 96047
 *
 * @param <T>
 */
public class SingleLinkedList<T> implements MyList<T> {
    //链表头
    private ListNode<T> first;
    //链表尾
    private ListNode<T> last;
    //存储元素个数
    private int size;

    @Override
    public void add(T element) {
        if (first == null){
            first = new ListNode<T>(element);
            last = first;
        }else{
            last.next = new ListNode<T>(element);
            last = last.next;
        }
        size++;
    }

    @Override
    public void delete(T element) {
        ListNode<T> p = first;
        ListNode<T> pre = null;
        while (p != null){
            if (p.data.equals(element)){
                //如果要删除节点是头节点
                if (p == first){
                    first = first.next;
                }else{
                    pre.next = p.next;
                }
                size--;
                break;
            }
            pre = p;
            p = p.next;
        }
    }

    @Override
    public void delete(int index) {
        if (index < 0 || index >= size){
            return;
        }
        ListNode<T> p = first;
        ListNode<T> pre = null;
        int i = 0;
        while (p != null){
            if (i == index){
                //如果要删除节点是头节点
                if (p == first){
                    first = first.next;
                }else{
                    pre.next = p.next;
                }
                size--;
                break;
            }
            pre = p;
            p = p.next;
            i++;
        }
    }

    @Override
    public void update(int index, T element) {
        if (index < 0 || index >= size){
            return;
        }

        ListNode<T> p = first;
        int i = 0;
        while (p != null){
            if (i == index){
                //更新数据
                p.data = element;
                break;
            }
            p = p.next;
            i++;
        }
    }

    @Override
    public boolean contains(T element) {
        return indexOf(element) >= 0;
    }

    @Override
    public T at(int index) {
        ListNode<T> p = first;
        int i = 0;
        while (p != null){
            if (i == index){
                //返回数据
                return p.data;
            }
            p = p.next;
            i++;
        }

        return null;
    }

    @Override
    public int indexOf(T element) {
        ListNode<T> p = first;
        int i = 0;
        while (p != null){
            if (p.data.equals(element)){
                //返回下标
                return i;
            }
            p = p.next;
            i++;
        }

        return -1;
    }

    @Override
    public int length() {
        return size;
    }

    @Override
    public String toString() {
        StringBuilder res = new StringBuilder("[");
        ListNode<T> p = first;
        while (p != null){
            res.append(p.data.toString() + ",");
            p = p.next;
        }
        res.deleteCharAt(res.length() - 1);
        res.append("]");

        return res.toString();
    }
}
双向链表
/**
 * 双向链表(有头,有尾)
 * @author 96047
 *
 * @param <T>
 */
public class DoubleLinkedList<T> implements MyList<T> {
    //链表头
    private ListNode<T> first = new ListNode<T>(null);
    //链表尾
    private ListNode<T> last = new ListNode<T>(null);
    //存储元素个数
    private int size;

    //初始化
    public DoubleLinkedList(){
        first.next = last;
        last.pre = first;
    }

    @Override
    public void add(T element) {
        ListNode<T> p = new ListNode<T>(element);
        last.pre.next = p;
        p.pre = last.pre;
        p.next = last;
        last.pre = p;
        size++;
    }

    @Override
    public void delete(T element) {
        ListNode<T> p = first.next;
        while (p != last){
            if (p.data.equals(element)){
                p.pre.next = p.next;
                p.next.pre = p.pre;
                p.next = null;
                p.pre = null;
                break;
            }
            p = p.next;
        }
        size--;
    }

    @Override
    public void delete(int index) {
        if (index < 0 || index >= size){
            return;
        }
        ListNode<T> p = first.next;
        int i = 0;
        while (p != last){
            if (i == index){
                p.pre.next = p.next;
                p.next.pre = p.pre;
                p.next = null;
                p.pre = null;
                size--;
                break;
            }
            p = p.next;
            i++;
        }
    }

    @Override
    public void update(int index, T element) {
        if (index < 0 || index >= size){
            return;
        }

        ListNode<T> p = first.next;
        int i = 0;
        while (p != last){
            if (i == index){
                //更新数据
                p.data = element;
                break;
            }
            p = p.next;
            i++;
        }
    }

    @Override
    public boolean contains(T element) {
        return indexOf(element) >= 0;
    }

    @Override
    public T at(int index) {
        if (index < 0 || index >= size){
            return null; 
        }

        ListNode<T> p = first.next;
        int i = 0;
        while (p != last){
            if (i == index){
                //返回数据
                return p.data;
            }
            p = p.next;
            i++;
        }

        return null;
    }

    @Override
    public int indexOf(T element) {
        ListNode<T> p = first.next;
        int i = 0;
        while (p != last){
            if (p.data.equals(element)){
                //返回下标
                return i;
            }
            p = p.next;
            i++;
        }

        return -1;
    }

    @Override
    public int length() {
        return size;
    }

    @Override
    public String toString() {
        StringBuilder res = new StringBuilder("[");
        ListNode<T> p = first.next;
        while (p != last){
            res.append(p.data.toString() + ",");
            p = p.next;
        }
        res.deleteCharAt(res.length() - 1);
        res.append("]");

        return res.toString();
    }

    private ListNode<T> now;    //当前节点

    //是否有下一个节点
    public boolean hasNext(){
        return now.next != last;
    }

    //得到当前节点数据并且指针后移
    public T next(){
        now = now.next;
        return now.pre.data;
    }
}

2.1 编写代码，移除未排序链表中的重复节点。

  • 解：

      class Node{
      Object data;
      Node next;
  }

  public static void main(String[] args) {
      int[] data = {1, 6, 7, 3, 7, 6};

      //构建链表
      Node head = new Node();
      head.data = data[0];
      Node p = head;
      for (int i = 1; i < data.length; i++) {
          Node node = new Node();
          node.data = data[i];
          p.next = node;
          p = node;
      }
      //去除重复元素
      removeRepeat(head);

      //输出
      p = head;
      while (p != null){
          System.out.print(p.data + " ");
          p = p.next;
      }
  }

  //去重
  public static void removeRepeat(Node head){
      Node p = head;    //遍历指针
      Node pre = null;    //前驱指针
      HashSet set = new HashSet();    //利用set判重
      while (p != null){
          if (set.contains(p.data)){    //如果set中已经有该元素了证明元素重复
              pre.next = p.next;    //删除该节点
          }else{
              set.add(p.data);    //将数据元加入set
              pre = p;
          }
          p = p.next;
      }
  }

2.2 实现一个算法，找出单向链表中倒数第k个节点。

  • 利用快慢指针

  • 解：

      //找到倒数第k个元素
  private static Node print(Node head, int k) {
      //数据校验
      if (head == null || k <= 0){
          return null;
      }
      int count = 1;
      //快指针走到顺数第三个元素
      Node p1 = head;
      while (p1 != null && count <= k){
          p1 = p1.next;
          count++;
      }

      //判断k是否在范围内
      if (p1 == null){
          return null;
      }

      //慢指针和快指针一起走，当快指针走完的时候慢指针正好走到倒数第k个位置
      Node p2 = head;
      while (p1 != null){
          p1 = p1.next;
          p2 = p2.next;
      }

      System.out.println(p2.data);
      return p2;
  }

public static void removeNode(Node node){
    //如果没有后继节点则返回
    if (node.next == null){
        return;
    }
    //因为无法回溯到前一节点删除，所以只能将后继节点数据拷贝后删除后继结点
    node.data = node.next.data;
    node.next = node.next.next;
}

2.4 以给定x为基准将链表分割成两部分，所有小于x的节点排在大于或等于x的节点之前。

private static Node p(Node head, int x) {
    Node high = null;    //大于等于x的头指针
    Node highP = null;    //大于等于x子链表的尾指针
    Node low = null;    //小于x的头指针
    Node lowP = null;    //小于x的尾指针
    Node p = head;
    while (p != null){
        if ((int)p.data < x){
            //如果当前节点的元素值小于x就将这个节点加入到low子链表中
            if (low == null){
                low = p;
                lowP = p;
            }else{
                lowP.next = p;
                lowP = lowP.next;
            }
        }else{
            //如果当前节点的元素值不小于x就将这个节点加入到high子链表中
            if (high == null){
                high = p;
                highP = p;
            }else{
                highP.next = p;
                highP = highP.next;
            }
        }
        p = p.next;
    }

    lowP.next = high;    //将两个子链表合并
    return low;
}

2.5 给定两个用链表表示的整数,每个结点包含-一个数位。这些数位是反向存放的，也就是个位排在链表首部。编写函数对这两个整数求和，并用链表形式返回结果。
示例
输入:(7->1->6)+(5->9->2)，即617+295。
输出:2->1->9，即912。
进阶
假设这些数位是正向存放的，请再做- -遍。
示例.
输入:(6->1->7)+(2->9->5)，即617+295。
输出:9->1->2，即912。

public static void main(String[] args) {
    Node num1 = new Node();
    num1.data = 6;
    Node num2 = new Node();
    num2.data = 1;
    Node num3 = new Node();
    num3.data = 7;
    num1.next = num2;
    num2.next = num3;
    Node num11 = new Node();
    num11.data = 2;
    Node num22 = new Node();
    num22.data = 9;
    Node num33 = new Node();
    num33.data = 5;
    num11.next = num22;
    num22.next = num33;

    Node h = sum(num1, num11, 0);

    while (h != null) {
        System.out.print(h.data);
        h = h.next;
    }
    System.out.println();
    NodeN nodeN = sum(num1, num11);
    h = nodeN.node;
    while (h != null) {
        System.out.print(h.data);
        h = h.next;
    }
}

//反向存放求和
public static Node sum(Node n1, Node n2, int n) {
    if (n1 == null && n2 == null && n == 0) {
        return null;
    }

    int val = n;    //val代表本位求和后的值 n为进位
    if (n1 != null) {
        val += n1.data;
    }
    if (n2 != null) {
        val += n2.data;
    }

    Node node = new Node();
    node.data = val % 10;
    //本位求得的结果链接上高位(也就是链表后继节点)求得的结果
    node.next = sum(n1 == null ? null : n1.next, 
            n2 == null ? null : n2.next, val >= 10 ? 1 : 0);

    return node;
}

//正向求和 不考虑非对其情况
public static NodeN sum(Node n1, Node n2){
    if (n1.next == null && n2.next == null){
        //如果本位为个位则求得相加结果并改变进位状态后返回
        int val = n1.data + n2.data;
        return new NodeN(new Node(val % 10), val >= 10);
    }

    //得到低位的结果
    NodeN nn = sum(n1.next, n2.next);
    int val = 0;
    //如果地位有进位则累加上
    if (nn.f){
        val += 1;
    }
    val += n1.data + n2.data;
    //将本位求得的结果和低位求得的结果链接起来
    Node node = new Node(val % 10);
    NodeN nodeN = new NodeN(node, val >= 10);
    nodeN.node.next = nn.node;

    return nodeN;
}

class NodeN{
    Node node;    //节点
    boolean f;    //进位

    public NodeN(Node node, boolean f){
        this.node = node;
        this.f = f;
    }
}

2.6 给定一个有环链表，实现-个算法返回环路的开头结点。有环链表的定义在链表中某个结点的next元素指向在它前面出现过的结点，则表明该链表存在环路。
示例
输人: A-> B -> C-> D-> E-> C (C结点出现了两次)。
输出: C

//返回链表的环路节点
public static Node loopNode(Node head){
    Node s = head;    //慢指针
    Node f = head;    //快指针

    //走到快慢指针的交汇点
    while (f != null && f.next != null){
        s = s.next;
        f = f.next.next;
        if (s == f){
            break;
        }
    }

    //如果链表没用环路则返回
    if (s != f){
        return null;
    }

    //head和快慢指针交汇点距离环路开始的点相同
    Node p = head;
    //找到环路开始的点
    while (p != s){
        p = p.next;
        s = s.next;
    }

    return p;
}

2.7 编写一个函数，检查链表是否为回文。

public static boolean isP(Node head){
    if (head == null){
        return false;
    }
    if (head.next == null){
        return true;
    }
    Node s = head;
    Node f = head;
    //利用快慢指针将s移动到中间节点
    while (f != null && f.next != null){
        s = s.next;
        f = f.next.next;
    }

    Node p = head;
    Stack<Integer> stack = new Stack<Integer>();
    //将前半部分元素压栈
    while (p != s){
        stack.push(p.data);
        p = p.next;
    }
    //s指向后半部分链表
    s = s.next;
    //比较前半部分和后半部分是否一致
    while (s != null){
        if (stack.pop() != s.data){
            return false;
        }
        s = s.next;
    }

    return true;
}

3.2 请设计-一个栈，除pop与push方法,还支持min方法，可返回栈元素中的最小值。push、pop和min三个方法的时间复杂度必须为0(1)。

public class _3_2_MyStackWithMin extends DoubleLinkedList<Integer> implements IStack<Integer> {

    @Override
    public void push(Integer element) {
        add(element);
        Integer peek = help.peek();

        if (peek == null){
            //如果最小值栈为空直接更新最小值
            help.add(element);
        }else{
            //判断如果入栈元素小于最小值则更新最小值
            if (element < peek){
                help.add(element);
            }
        }
    }

    @Override
    public Integer pop() throws EmptyStackException {
        if (size <= 0){
            throw new EmptyStackException();
        }
        ListNode<Integer> t = last.pre;
        t.pre.next = last;
        last.pre = t.pre;
        t.next = null;
        t.pre = null;
        size--;

        Integer peek = help.peek();
        // 如果出栈元素小于等于最小值则最小值需要出栈进行更新
        if (t.data <= peek) {
            help.pop();
        }

        return t.data;
    }

    @Override
    public boolean isEmpty() {
        return size == 0;
    }

    @Override
    public int size() {
        return size;
    }

    @Override
    public Integer peek() {
        return last.pre.data;
    }
    //维护栈最小值的辅助栈
    private MyStack<Integer> help = new MyStack<>();

    public Integer min(){
        return help.peek();
    }

    public static void main(String[] args) {
        _3_2_MyStackWithMin sm = new _3_2_MyStackWithMin();
        sm.push(1);
        sm.push(3);
        sm.push(2);
        sm.push(4);
        sm.push(1);

        System.out.println(sm.min());
        sm.pop();
        System.out.println(sm.min());
    }
}

3.3 设想有一堆盘子,堆太高可能会倒下来。因此，在现实生活中，盘子堆到一定高度时，我们就会另外堆-堆盘子。请实现数据结构SetOfStacks,模拟这种行为。Set0fStacks应 该由多个栈组成，并且在前一个栈填满时新建一个栈。此外，SetOfStacks . push()和SetOfStacks . pop( )应该与普通栈的操作方法相同(也就是说，pop( )返回的值，应该跟只有一个栈时的情况- -样)。
进阶
实现一个popAt(int index)方法， 根据指定的子栈，执行pop操作。

public class _3_3_SetOfStacks {
    //总栈
    private ArrayList<ArrayList<Integer>> stacks = new  ArrayList<ArrayList<Integer>>();
    //当前栈
    private ArrayList<Integer> curStack = new ArrayList<Integer>();
    //当前栈容量上限
    private int size = 5;

    public _3_3_SetOfStacks(){
        stacks.add(curStack);
    }

    public void push(int e){
        //如果当前栈还有容量则直接入栈
        if (curStack.size() < size){
            curStack.add(e);
        }else{
            //如果当前栈已满则创建新的栈加入元素，并且将当前栈加入总栈
            curStack = new ArrayList<Integer>();
            curStack.add(e);
            stacks.add(curStack);
        }
    }

    public int pop(){
        //判断当前栈是否还能弹出
        if (curStack.isEmpty()){
            //不能弹出则删除这个空栈
            stacks.remove(stacks.size() - 1);
            //如果删除当前栈后总栈为空则直接返回
            if (stacks.isEmpty()){
                return -1;
            }
            //在总栈中取得新的当前栈
            curStack = stacks.get(stacks.size() - 1);
            //弹出元素
            return curStack.remove(curStack.size() - 1);
        }else{
            //弹出元素
            return curStack.remove(curStack.size() - 1);
        }
    }

    //根据指定子栈返回元素
    public int popAt(int index){
        //判断下标是否越界
        if (index < 0 || index >= stacks.size()){
            return -1;
        }
        //得到指定的栈
        ArrayList<Integer> cur = stacks.get(index);
        //判断栈是否为空
        if (cur.isEmpty()){    //如果栈为空
            //删除该栈
            stacks.remove(index);
            //重新返回该位置有元素的栈中的元素
            return popAt(index);
        }else{
            //如果不为空返回元素
            return cur.remove(cur.size() - 1);
        }
    }

    public static void main(String[] args) {
        _3_3_SetOfStacks stacks = new _3_3_SetOfStacks();
        stacks.push(1);
        stacks.push(2);
        stacks.push(3);
        stacks.push(4);
        stacks.push(5);
        stacks.push(6);
        stacks.push(7);

        System.out.println(stacks.popAt(0));
    }
}

3.5 实现一个MyQueue类，该类用两个栈来实现-一个队列。

public class _3_5_MyQueue {
    //主栈
    private Stack<Integer> p = new Stack<>();
    //辅助栈
    private Stack<Integer> s = new Stack<>();

    public void enqueue(int e){
        //如果主栈为空则将元素移到主栈
        if (p.empty()){
            move(s, p);
        }
        p.push(e);
    }

    public Integer dequeue(){
        //如果辅助栈为空则从主栈移入
        if (s.empty()){
            move(p, s);
        }
        return s.pop();
    }
    //将栈中元素从src移动到tar
    private void move(Stack<Integer> src, Stack<Integer> tar){
        while (!src.empty()){
            tar.push(src.pop());
        }
    }

    public static void main(String[] args) {
        _3_5_MyQueue q = new _3_5_MyQueue();
        q.enqueue(1);
        q.enqueue(2);
        q.enqueue(3);
        q.enqueue(4);

        System.out.println(q.dequeue());
        q.enqueue(5);
        System.out.println(q.dequeue());
        System.out.println(q.dequeue());
        System.out.println(q.dequeue());
        System.out.println(q.dequeue());
    }
}

3.6 对栈进行排序，最多只能使用一个额外的栈存放临时数据。

public static void stackSort(int[] nums){
    //将传入数据恢复成栈
    Stack<Integer> stack = new Stack<Integer>();
    for (int i = nums.length - 1; i >= 0 ; i--) {
        stack.push(nums[i]);
    }
    //创建辅助栈
    Stack<Integer> help = new Stack<Integer>();
    //将数据按有序放入辅助栈
    while (!stack.empty()){
        //弹出一个元素
        int t = stack.pop();
        //如果辅助栈没有数据
        if (help.empty()){
            //直接放入
            help.push(t);
        }else{
            //否则将比当前元素大的元素先回填到主栈再将元素放入
            while (!help.empty() && t < help.peek()){
                stack.push(help.pop());
            }
            help.push(t);
        }
    }
    //输出排序后的结果
    while (!help.empty()){
        System.out.print(help.pop() + " ");
    }
    System.out.println();
}

3.7 有家动物收容所只收留猫和狗，但有特殊的收养规则，收养人有两种收养方式，
第一种为直接收养所有动物中最早进入收容所的，
第二种为选择收养的动物类型（猫或狗），并收养该种动物中最早进入收容所的。

给定一个操作序列int[][2] ope(C++中为vector<vector>)代表所有事件。
若第一个元素为1，则代表有动物进入收容所，第二个元素为动物的编号，正数代表狗，负数代表猫；
若第一个元素为2，则代表有人收养动物，第二个元素若为0，则采取第一种收养方式(最早)，
若为1，则指定收养狗，若为-1则指定收养猫。
请按顺序返回收养的序列。若出现不合法的操作，即没有可以符合领养要求的动物，则将这次领养操作忽略。
测试样例：

[[1,1],[1,-1],[2,0],[2,-1]]

返回：[1,-1]

public static List<Integer> adopt(int[][] ope) {
    //狗队列
    Queue<Animal> dog = new LinkedList<>();
    //猫队列
    Queue<Animal> cat = new LinkedList<>();
    //收养结果
    List<Integer> res = new ArrayList<>();
    for (int[] i : ope) {
        int opt = i[0]; // 得到操作动作
        int type = i[1];// 得到操作类型
        if (opt == 1) {
            // 有动物进入
            if (type > 0) {
                // 正数为狗
                dog.add(new Animal(type));
            } else if (type < 0) {
                // 负数为猫
                cat.add(new Animal(type));
            }
        } else if (opt == 2) {
            // 有人收养
            if (type == 1 && !dog.isEmpty()) {
                // 收养狗且狗不为空则收养成功
                res.add(dog.poll().type);
            } else if (type == -1 && !cat.isEmpty()) {
                // 收养猫且猫不为空则收养成功
                res.add(cat.poll().type);
            } else if (type == 0) {
                // 收养进入时间最早的
                int tDog = Integer.MAX_VALUE; // 将狗的时间初始化为最迟
                if (!dog.isEmpty()) {
                    // 如果还有狗则更新收养时间
                    tDog = dog.peek().time;
                }

                int tCat = Integer.MAX_VALUE; // 将猫的收养时间初始化为最迟
                if (!cat.isEmpty()) {
                    // 如果还有猫则更新
                    tCat = cat.peek().time;
                }
                if (tDog < tCat) {
                    // 如果狗的收养时间早则收养成功
                    res.add(dog.poll().type);
                }
                if (tDog > tCat) {
                    // 如果猫的收养时间早则收养成功
                    res.add(cat.poll().type);
                }
            }
        }
    }

    return res;
}

public static void main(String[] args) {
    int[][] data = { { 1, 1 }, { 1, -1 }, { 1, 0 }, { 1, -5 }, { 2, 0 }, { 2, -1 }, {2, 0}};
    System.out.println(adopt(data));
}
