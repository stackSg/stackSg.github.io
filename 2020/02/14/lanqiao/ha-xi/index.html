public interface IMap<K, V> {
    //清空map
    void clear();

    //key是否已经存在
    boolean containsKey(K key);

    //value是否存在
    boolean containsValue(V value);

    //根据key获得value
    V get(K key);

    //map是否为空
    boolean isEmpty();

    //返回key集合
    ISet<K> keySet();

    //所有的value集合
    V[] values();

    //存入键值对
    void put(K key, V value);

    //将另一个map中的内容复制到当前map
    void putAll(IMap<? extends K, ? extends V> map);

    //根据key删除键值对
    V remove(K key);

    //map大小
    int size();

    //返回迭代器
    Iterator<Node<K, V>> iterator();
}
  • 实现
public class MyHashMap<K, V> implements IMap<K, V> {
    //容量
    private int length = 16;
    //map的容器
    private Node[] buckets = new Node[length];
    //map大小
    private int size;

    @Override
    public void clear() {
        for (int i = 0; i < buckets.length; i++) {
            buckets[i] = null;
        }
        size = 0;
    }

    @Override
    public boolean containsKey(K key) {
        //得到key的hash值
        int index = hash(key);
        //如果该桶为空直接返回
        if (buckets[index] == null){
            return false;
        }
        //遍历桶匹配key值
        Node<K, V> node = buckets[index];
        while (node != null){
            if (node.key == key || node.key.hashCode() == key.hashCode() && node.key.equals(key)){
                return true;
            }
            node = node.next;
        }

        return false;
    }

    @Override
    public boolean containsValue(V value) {
        //遍历桶匹配value值
        for (int i = 0; i < buckets.length; i++) {
            Node<K, V> node = buckets[i];
            while (node != null){
                if (node.value.equals(value)){
                    return true;
                }
                node = node.next;
            }
        }
        return false;
    }

    @Override
    public V get(K key) {
        //得到key的hash值
        int index = hash(key);
        //如果该桶为空直接返回
        if (buckets[index] == null){
            return null;
        }
        //遍历桶匹配key值
        Node<K, V> node = buckets[index];
        while (node != null){
            if (node.key == key || node.key.hashCode() == key.hashCode() && node.key.equals(key)){
                return node.value;
            }
            node = node.next;
        }

        return null;
    }

    @Override
    public boolean isEmpty() {
        return size == 0;
    }

    @Override
    public ISet<K> keySet() {
        ISet<K> set = new MyHashSet<>();
        for (int i = 0; i < buckets.length; i++) {
            Node<K, V> temp = buckets[i];
            while (temp != null){
                set.add(temp.key);
                temp = temp.next;
            }
        }
        return set;
    }

    @Override
    public V[] values() {
        // TODO Auto-generated method stub
        return null;
    }

    @Override
    public void put(K key, V value) {
        Node<K, V> temp = new Node<K, V>(key, value);
        int index = hash(key);
        Node<K, V> node = buckets[index];
        //判断该桶中是否有元素
        if (node == null){
            //如果为空直接放入该元素
            buckets[index] = temp;
        }else{
            //遍历链表到末尾
            while (node != null){
                //判重
                if (node.key == key || node.hashCode() == temp.hashCode() && node.key.equals(temp.key)){
                    //如有重复更新value值
                    node.value = value;
                    return;
                }
                node = node.next;
                if (node.next == null){
                    break;
                }
            }

            //将该元素添加到链表末尾
            node.next = temp;
        }
        size++;
    }

    private int hash(K key) {
        return key.hashCode() % 16;
    }

    @Override
    public void putAll(IMap<? extends K, ? extends V> map) {
        // TODO Auto-generated method stub

    }

    @Override
    public V remove(K key) {
        //得到key的hash值
        int index = hash(key);
        //如果该桶为空直接返回
        if (buckets[index] == null){
            return null;
        }
        //遍历桶匹配key值
        Node<K, V> node = buckets[index];
        Node<K, V> pre = node;
        while (node != null){
            if (node.key == key || node.key.hashCode() == key.hashCode() && node.key.equals(key)){
                if (node == pre){
                    buckets[index] = null;
                }else{
                    pre.next = node.next;
                }
                size--;
                return node.value;
            }
            pre = node;
            node = node.next;
        }

        return null;
    }

    @Override
    public int size() {
        return size;
    }
    @Override
    public Iterator<Node<K, V>> iterator(){
        return new Iterator<Node<K, V>>() {
            private int i;
            private Node<K, V> p = buckets[i];
            @Override
            public boolean hasNext() {
                while (i < length && p == null){
                    i++;
                    if (i == 16){
                        break;
                    }
                    p = buckets[i];
                }
                return p != null;
            }

            @Override
            public Node<K, V> next() {
                Node<K, V> node = p;
                p = p.next;
                return node;
            }
        };
    }
    @Override
    public String toString() {
        StringBuilder res = new StringBuilder("[");
        for (int i = 0; i < buckets.length; i++) {
            Node<K, V> p = buckets[i];
            while (p != null){
                res.append("[" + p.key + "," + p.value + "],");
                p = p.next;
            }
        }
        res.append("]");

        return res.toString();
    }
}

class Node<K, V>{
    K key;
    V value;

    Node<K, V> next;

    public Node(K key, V value) {
        super();
        this.key = key;
        this.value = value;
    }
}

HashSet

  • 操作定义
public interface ISet<E> {
    void add(E e);

    E remove(E e);

    void clear();

    boolean contains(E e);

    boolean isEmpty();

    int size();

    Iterator<E> iterator();
}
  • 实现
public class MyHashSet<E> implements ISet<E> {
    //持有hashmap实例
    private MyHashMap<E, E> set = new MyHashMap<>();

    @Override
    public void add(E e) {
        set.put(e, null);
    }

    @Override
    public E remove(E e) {
        return set.remove(e);
    }

    @Override
    public void clear() {
        set.clear();
    }

    @Override
    public boolean contains(E e) {
        return set.containsKey(e);
    }

    @Override
    public boolean isEmpty() {
        return set.isEmpty();
    }

    @Override
    public int size() {
        return set.size();
    }

    @Override
    public Iterator<E> iterator() {
        return new Iterator<E>() {
            Iterator<Node<E, E>> it = set.iterator();
            @Override
            public boolean hasNext() {
                return it.hasNext();
            }

            @Override
            public E next() {
                return (E) it.next().key;
            }
        };
    }

    @Override
    public String toString() {
        StringBuilder res = new StringBuilder("[");
        Iterator<E> it = iterator();
        while (it.hasNext()){
            E e = it.next();
            res.append(e.toString() + ",");
        }
        res.append("]");
        return res.toString();
    }

}

布隆过滤器

  • 布隆过滤器(英语: Bloom Filter )是1970年由布隆提出的。它实际上是一个很长的二进制向量和一系列随机映射函数。布隆过滤器可以用于检索一个元素是否在一个集合中。 它的优点是空间效率和查询时间都远远超过一般的算法,缺点是有一定的误识别率和删除困难。

  • 如果想判断一个元素是不是在一个集合里,-般想到的是将集合中所有元素保存起来,然后通过比较确定。链表、树、散列表(又叫哈希表, Hash table )等等数据结构都是这种思路。但是随着集合中元素的增加,我们需要的存储空间越来越大。同时检索速度也越来越慢,上述三种结构的检索时间复杂度分别为O(n),O(logn),O(n/k)。布隆过滤器的原理是,当-个元素被加入集合时,通过K个散列函数将这个元素映射成一个位数组中的K个点 ,把它们置为1。检索时,我们只要看看这些点是不是都是1就(大约)知
    道集合中有没有它了:如果这些点有任何一个0 ,则被检元素- -定不在;如果都是1 ,则被检元素很可能在。这就是布隆过滤器的基本思想。

  • 实现

/**
 * 布隆过滤器
 * @author 96047
 *
 */
public class BloomFilter {
    //位图的长度
    private static final int NUM_SLOTS = 1024 * 1024 * 8;
    //hash函数的个数
    private static final int NUM_HASH = 8;
    //位图
    private BigInteger bitMap = new BigInteger("0");

    //将message+n映射到0~NUM_SLOTS-1之间的一个值
    private int hash(String message, int n){
        try {
            MessageDigest md5 = MessageDigest.getInstance("md5");
            message = message + String.valueOf(n);
            byte[] bytes = message.getBytes();
            md5.update(bytes);
            byte[] digst = md5.digest();
            BigInteger b = new BigInteger(digst);

            return Math.abs(b.intValue()) % NUM_SLOTS;
        } catch (NoSuchAlgorithmException e) {
            e.printStackTrace();
        }

        return -1;
    }
    //添加元素
    public void addElement(String message){
        for (int i = 0; i < NUM_HASH; i++){
            int hash = hash(message, i);
            if (!bitMap.testBit(hash)){
                bitMap = bitMap.or(new BigInteger("1").shiftLeft(hash));
            }
        }
    }
    //检查元素是否在位图中
    public boolean check(String message){
        for (int i = 0; i < NUM_HASH; i++){
            int hash = hash(message, i);
            if (!bitMap.testBit(hash)){
                return false;
            }
        }

        return true;
    }
}

一致性Hash

  • 先构造一个长度为232的整数环(这个环被称为一致性Hash环) ,根据节点名称的Hash值(其分布为[0, 232-1] )将缓存服务器节点
    放置在这个Hash环上,然后根据需要缓存的数据的Key值计算得到其Hash值(其分布也为[0, 232-1] ) , 然后在Hash环上顺时针查
    找距离这个Key值的Hash值最近的服务器节点,完成Key到服务器的映射查找。
  • 增加/删除节点：如果往集群中添加一个新的节点NODE4 ,通过对应的哈希算法得到KEY4 ,并映射到环中按顺时针迁移的规则,那么被分割的对象被迁移到了NODE4中,其它对象还保持这原有的存储位置。
  • 数据倾斜.如果机器较少,很有可能造成机器在整个环上的分布不均匀,从而导致机器之间的负载不均衡

题解

  1. 有一个包含20亿个全是32位整数的大文件，在其中找到出现次数最多的数
    • 使用hash将这个大文件分散为多个小文件，扫描每个小文件得到出现次数最多的数，然后比较每个小文件中出现次数最多的数。
  2. 32位无符号整数的范围是0~4294967295，现在有-一个正好包含40亿个无符号整数的文件，所以在整个范围中必然有没出现过的数。可以使用最多1G的内存，怎么找到所有没出现过的数。
    • 使用bitmap+下标索引做统计。
  3. 找到100亿个URL中重复的URL以及搜索词汇的topK问题。
    • hash分流–>每个小文件中统计词频，同时维护top100的小顶堆。–>n个小顶堆，归并排序，外部排序
