树的实现

操作定义
/**
 * 树接口
 * @author 96047
 *
 * @param <E>
 */
public interface ITree<E> {
    //节点数
    int size();

    //获取根节点
    TreeNode<E> getRoot();

    //获取n的父节点
    TreeNode<E> getParent(TreeNode<E> n);

    //获取第一个儿子
    TreeNode<E> getFirstChild(TreeNode<E> p);

    //获取n的下一个兄弟
    TreeNode<E> getNextSibling(TreeNode<E> n);

    //子树高度
    int getHight(TreeNode<E> n);

    //插入子节点
    void insertChild(TreeNode<E> n, TreeNode<E> child);

    //删除第i个子节点
    void deleteChild(TreeNode<E> n, int i);

    //先序遍历
    List<TreeNode<E>> preOrder(TreeNode<E> n);

    //后序遍历
    List<TreeNode<E>> postOrder(TreeNode<E> n);

    //层序遍历
    List<List<TreeNode<E>>> levelOrder(TreeNode<E> n);

    List<List<TreeNode<E>>> levelOrder();
}

实现
public class MyTree<E> implements ITree<E> {
    private TreeNode<E> root;
    private int size;

    public MyTree(TreeNode<E> root){
        this.root = root;
        size++;
    }

    @Override
    public int size() {
        return size;
    }

    @Override
    public TreeNode<E> getRoot() {
        return root;
    }

    @Override
    public TreeNode<E> getParent(TreeNode<E> n) {
        if (n == null){
            return null;
        }else{
            return n.parent;
        }
    }

    @Override
    public TreeNode<E> getFirstChild(TreeNode<E> p) {
        if (p == null || p.children == null){
            return null;
        }else{
            return p.children.get(0);
        }
    }

    @Override
    public TreeNode<E> getNextSibling(TreeNode<E> n) {
        if (n == null || n.parent == null){
            return null;
        }else{
            int index = n.parent.children.indexOf(n);
            TreeNode<E> t = n.parent.children.get(index + 1);
            return t;
        }
    }

    @Override
    public int getHight(TreeNode<E> n) {
        if (n.children == null){
            return 0;
        }
        int h = 0;
        for (TreeNode<E> t : n.children){
            h = Math.max(h, getHight(t));
        }

        return h + 1;
    }

    @Override
    public void insertChild(TreeNode<E> n, TreeNode<E> child) {
        if (n.children == null){
            n.children = new ArrayList<TreeNode<E>>();
        }
        n.children.add(child);
        child.parent = n;
        size++;
    }

    @Override
    public void deleteChild(TreeNode<E> n, int i) {
        n.children.remove(i);
        size--;
    }

    @Override
    public List<TreeNode<E>> preOrder(TreeNode<E> n) {
        if (n.children == null){
            List<TreeNode<E>> list = new ArrayList<TreeNode<E>>();
            list.add(n);

            return list;
        }
        List<TreeNode<E>> list = new ArrayList<TreeNode<E>>();
        //遍历根
        list.add(n);
        //得到子层遍历结果
        List<TreeNode<E>> listC = null;
        for (TreeNode<E> t : n.children){
            listC = preOrder(t);
            list.addAll(listC);
        }

        return list;
    }

    @Override
    public List<TreeNode<E>> postOrder(TreeNode<E> n) {
        if (n.children == null){
            List<TreeNode<E>> list = new ArrayList<TreeNode<E>>();
            list.add(n);
            return list;
        }
        //得到子层遍历结果
        List<TreeNode<E>> list = new ArrayList<TreeNode<E>>();
        for (TreeNode<E> t : n.children){
            list.addAll(postOrder(t));
        }
        //加入根
        list.add(n);

        return list;
    }

    @Override
    public List<List<TreeNode<E>>> levelOrder(TreeNode<E> n) {
        List<List<TreeNode<E>>> res = new ArrayList<List<TreeNode<E>>>();
        List<TreeNode<E>> list = new ArrayList<TreeNode<E>>();
        //初始化队列
        Queue<TreeNode<E>> queue = new LinkedList<TreeNode<E>>();
        queue.add(n);
        //双指针记录层级关系
        TreeNode<E> p = n;    //游标指针
        TreeNode<E> last = n;    //该层最后一个元素

        while (!queue.isEmpty()){
            //弹出一个
            TreeNode<E> t = queue.poll();
            list.add(t);

            //加入n个
            if (t.children != null){
                for (TreeNode<E> tn : t.children){
                    queue.add(tn);
                    p = tn;    //游标指针指向新加入队列的最后一个元素
                }
            }
            //如果弹出的元素是该层最后一个元素
            if (t == last){
                last = p;    //last指向下一层最后一个元素
                res.add(list);    //该层遍历结果加入总结果集
                list = new ArrayList<TreeNode<E>>();    //初始化新一层结果集
            }
        }

        return res;
    }

    @Override
    public List<List<TreeNode<E>>> levelOrder() {
        return levelOrder(root);
    }

}

二叉树的性值

  1. 在非空二叉树中，第i层的结点总数不超过 , i>=1；
  2. 深度为h的二叉树最多有 个结点(h>=1)，最少有h个结点；
  3. 对于任意一棵二叉树，如果其叶结点数为N0，而度数为2的结点总数为N2，则N0=N2+1；
  4. 具有n个结点的完全二叉树的深度为 （注：[ ]表示向下取整）
  5. 有N个结点的完全二叉树各结点如果用顺序方式存储，则结点之间有如下关系：
    若I为结点编号则 如果I>1，则其父结点的编号为I/2；
    如果2I<=N，则其左孩子（即左子树的根结点）的编号为2I；若2I>N，则无左孩子；
    如果2    I+1<=N，则其右孩子的结点编号为2I+1；若2I+1>N，则无右孩子。

public interface IBinarySearchTree<K, V> {
    //插入一个节点
    BSTNode<K, V> insert(K key, V value);

    //中序遍历
    void inOrder(Consumer<K> con);

    //查找value
    V lookUpValue(K key);

    //获取最大关键字
    K max();

    //获取最小关键字
    K min();

    //删除节点
    void remove(K key);

    //第一个比k大的节点
    K successor(K key);

    //前一个比K小的节点
    K predecessor(K key);

    //二叉树是否平衡
    boolean isBalance();

    //节点数
    int getSize();

    //高度
    int getHeight();

    //层序遍历
    List<List<BSTNode<K, V>>> levelOrder();
}

节点定义

public class BSTNode<K, V> {
    public K key;
    public V vlaue;
    public BSTNode<K, V> left;
    public BSTNode<K, V> right;
    public BSTNode<K, V> parent;
    public boolean isLeftChild;
    public int height;
    public int num;
    public boolean isRed = true;

    public BSTNode(K key, V vlaue, BSTNode<K, V> left, BSTNode<K, V> right, BSTNode<K, V> parent) {
        super();
        this.key = key;
        this.vlaue = vlaue;
        this.left = left;
        this.right = right;
        this.parent = parent;
    }
    public BSTNode(K key, V vlaue) {
        super();
        this.key = key;
        this.vlaue = vlaue;
    }
    @Override
    public String toString() {
        StringBuilder res = new StringBuilder();
        if (isRed){
            res.append("[" + key + "]");
        }else{
            res.append("(" + key + ")");
        }
        if (parent != null){
            res.append("<--" + parent.key);
        }
        return res.toString();
    }
}

实现

public class BinarySearchTree<K, V> implements IBinarySearchTree<K, V> {
    protected BSTNode<K, V> root;
    protected int size;
    protected Comparator<K> comparator;

    public BinarySearchTree(BSTNode<K, V> root) {
        super();
        this.root = root;
    }

    @Override
    public BSTNode<K, V> insert(K key, V value) {
        if (!(key instanceof Comparable)) {
            throw new ClassCastException();
        }
        BSTNode<K, V> p = null;
        BSTNode<K, V> cur = root;
        // 找到应该插入的位置的父节点
        while (cur != null) {
            p = cur;
            if (compare(cur.key, key) < 0) {
                cur = cur.right;
            } else if (compare(cur.key, key) > 0) {
                cur = cur.left;
            } else {
                cur.vlaue = value;
                return cur;
            }
        }
        cur = new BSTNode<K, V>(key, value);
        // 如果树为空
        if (p == null) {
            root = cur;
        } else if (compare(p.key, key) > 0) {
            p.left = cur;
            cur.isLeftChild = true;
        } else {
            p.right = cur;
            cur.isLeftChild = false;
        }
        cur.parent = p;
        size++;
        updateHight(cur);
        return cur;
    }

    protected void updateHight(BSTNode<K, V> cur) {
        if (cur.parent == null) {
            return;
        }
        BSTNode<K, V> p = cur.parent;
        if (cur.height == p.height) {
            p.height++;
            updateHight(p);
        }
    }

    @Override
    public void inOrder(Consumer<K> con) {
        if (root != null) {
            inOrder(root, con);
        }
    }

    protected void inOrder(BSTNode<K, V> cur, Consumer<K> con) {
        if (cur == null)
            return;
        inOrder(cur.left, con);
        con.accept(cur.key);
        inOrder(cur.right, con);
    }

    @Override
    public V lookUpValue(K key) {
        return lookupNode(key).vlaue;
    }

    protected BSTNode<K, V> lookupNode(K key) {
        BSTNode<K, V> cur = root;
        while (cur != null) {
            if (compare(cur.key, key) == 0) {
                return cur;
            } else if (compare(cur.key, key) > 0) {
                cur = cur.left;
            } else {
                cur = cur.right;
            }
        }
        return null;
    }

    @Override
    public K max() {
        return maxNode(root).key;
    }

    protected BSTNode<K, V> maxNode(BSTNode<K, V> p) {
        while (p.right != null)
            p = p.right;
        return p;
    }

    @Override
    public K min() {
        return minNode(root).key;
    }

    protected BSTNode<K, V> minNode(BSTNode<K, V> p) {
        while (p.left != null)
            p = p.left;
        return p;
    }

    @Override
    public void remove(K key) {
        BSTNode<K, V> p = lookupNode(key);
        // 没找到
        if (p == null) {
            return;
        }
        // 只有一个根节点
        if (p.parent == null) {
            root = null;
            size--;
            return;
        }
        // 叶子节点
        if (p.left == null && p.right == null) {
            if (p.isLeftChild) {
                p.parent.left = null;
            } else {
                p.parent.right = null;
            }
            p.parent = null;
            p = null;
            size--;
            return;
        }
        // 有一个孩子
        if (p.left == null) {
            if (p.isLeftChild) {
                p.parent.left = p.right;
                p.isLeftChild = true;
                p.right.parent = p.parent;
            } else {
                p.parent.right = p.right;
                p.right.parent = p.parent;
            }
            p.parent = null;
            p = null;
            size--;
            return;
        }
        if (p.right == null) {
            if (p.isLeftChild) {
                p.parent.left = p.left;
                p.isLeftChild = true;
            } else {
                p.parent.right = p.left;
                p.isLeftChild = false;
            }
            p.left.parent = p.parent;
            p.parent = null;
            p = null;
            size--;
            return;
        }
        // 有两个孩子
        BSTNode<K, V> minOfRight = minNode(p.right);
        p.key = minOfRight.key;
        p.vlaue = minOfRight.vlaue;
        remove(minOfRight.key);
    }

    @Override
    public K successor(K key) {
        BSTNode<K, V> p = lookupNode(key);
        if (p == null)    return null;
        if (p.right != null){
            //右子树中最小的元素
            return minNode(p.right).key;
        }
        //向上找第一个是左孩子的节点的父节点
        while (p != null && !p.isLeftChild){
            p = p.parent;
        }
        return p == null ? null:p.parent.key;
    }

    @Override
    public K predecessor(K key) {
        BSTNode<K, V> p = lookupNode(key);
        if (p == null)    return null;
        if (p.left != null){
            //左子树中最大的元素
            return maxNode(p.left).key;
        }
        //向上找第一个是右孩子的节点的父节点
        while (p != null && p.isLeftChild){
            p = p.parent;
        }
        return p == null ? null:p.parent.key;
    }

    @Override
    public boolean isBalance() {
        return !unBalance(root);
    }

    protected boolean unBalance(BSTNode<K, V> n){
        if (n == null) return false;
        int x = Math.abs(getHight(n.left) - getHight(n.right));
        return x > 1 || unBalance(n.left) || unBalance(n.right);
    }

    @Override
    public int getSize() {
        return size;
    }

    @Override
    public int getHeight(){
        return getHight(root);
    }
    protected int getHight(BSTNode<K, V> p) {
        if (p == null){
            return 0;
        }
        int max = Math.max(getHight(p.left), getHight(p.right));
        return max + 1;
    }

    protected void rightRotate(BSTNode<K, V> g, BSTNode<K, V> p) {
        BSTNode<K, V> gp = g.parent;
        boolean gb = g.isLeftChild;

        g.parent = p;
        g.isLeftChild = false;


        //处理B
        if (p.right != null){
            g.left = p.right;
            g.left.isLeftChild = true;
            g.left.parent = g;
        }else{
            g.left = null;
        }
        p.right = g;

        //处理p
        if (gp != null){
            p.parent = gp;
            p.isLeftChild = gb;
            //处理上一级
            if (gb){
                p.parent.left = p;
            }else{
                p.parent.right = p;
            }
        }else{
            root = p;
            p.parent = null;
        }
    }

    protected void leftRotate(BSTNode<K, V> g, BSTNode<K, V> p) {
        BSTNode<K, V> gp = g.parent;
        boolean gb = g.isLeftChild;

        g.parent = p;
        g.isLeftChild = true;


        //处理B
        if (p.left != null){
            g.right = p.left;
            g.right.isLeftChild = false;
            g.right.parent = g;
        }else{
            g.right = null;
        }
        p.left = g;

        //处理p
        if (gp != null){
            p.parent = gp;
            p.isLeftChild = gb;
            //处理上一级
            if (gb){
                p.parent.left = p;
            }else{
                p.parent.right = p;
            }
        }else{
            root = p;
            p.parent = null;
        }
    }

    @Override
    public List<List<BSTNode<K, V>>> levelOrder() {
        BSTNode<K, V> p = root;
        List<List<BSTNode<K, V>>> res = new ArrayList<>();
        List<BSTNode<K, V>> list = new ArrayList<BSTNode<K,V>>();
        Queue<BSTNode<K, V>> queue = new LinkedList<BSTNode<K,V>>();
        queue.add(p);
        BSTNode<K, V> last = p;
        BSTNode<K, V> n = p;

        while (!queue.isEmpty()){
            //弹出一个
            p = queue.poll();
            list.add(p);
            //进入n个
            if (p.left != null){
                p.left.num = p.num * 2;
                queue.add(p.left);
                n = p.left;
            }
            if (p.right != null){
                p.right.num = p.num * 2 + 1;
                queue.add(p.right);
                n = p.right;
            }
            if (p == last){
                res.add(list);
                list = new ArrayList<>();
                last = n;
            }
        }

        return res;
    }

    protected int compare(K key1, K key2) {
        // 判断是否有自定义比较器
        if (comparator == null) {
            return ((Comparable) key1).compareTo((Comparable) key2);
        } else {
            return comparator.compare(key1, key2);
        }
    }
}

二叉平衡树

  • AVL是最先发明的自平衡二叉查找树
  • AVL树所有节点的平衡因子(高度最大差)的绝对值不大于1，需要在BST的基础上在插入和删除时维护平衡即可。
  • AVL is a BST.

public class AVLTree<K, V> extends BinarySearchTree<K, V> implements IBinarySearchTree<K, V> {
    public AVLTree(BSTNode<K, V> root) {
        super(root);
    }

    @Override
    public BSTNode<K, V> insert(K key, V value) {
        //插入节点
        BSTNode<K, V> node = super.insert(key, value);
        //找到不平衡的节点坐标
        BSTNode<K, V>[] gps = findUnBalance(node);
        if (gps != null){
            //重新调整平衡
            reBalance(gps);
        }
        return node;
    }
    //根据gps将二叉树调整平衡
    private void reBalance(BSTNode<K, V>[] gps) {
        if (gps == null) return;
        BSTNode<K, V> g = gps[0];
        BSTNode<K, V> p = gps[1];
        BSTNode<K, V> s = gps[2];

        //判型
        if (p.isLeftChild && s.isLeftChild){    //左左型
            rightRotate(g, p);
        }else if (!p.isLeftChild && !s.isLeftChild){ //右右型
            leftRotate(g, p);
        }else if (p.isLeftChild && !s.isLeftChild){    //左右型
            leftRotate(p, s);
            rightRotate(g, s);
        }else{    //右左型
            rightRotate(p, s);
            leftRotate(g, s);
        }
    }

    //向上寻找到第一个不平衡的祖先
    private BSTNode<K, V>[] findUnBalance(BSTNode<K, V> node) {
        if (node == null) return null;
        BSTNode<K, V> s = node;
        if (node.parent == null) return null;
        BSTNode<K, V> p = node.parent;
        if (p.parent == null) return null;
        BSTNode<K, V> g = p.parent;
        if (unBalance(g)){
            return new BSTNode[]{g, p, s};
        }else{
            return findUnBalance(p);
        }
    }

    @Override
    public void remove(K key) {
        BSTNode<K, V> node = lookupNode(key);
        if (node == null)     return;

        if (node.left == null && node.right == null){
            //如果是叶子节点直接删除后重新调整平衡
            super.remove(key);
            reBalance(node.parent);
        }else if (node.left == null){
            //如果只有右孩子,在右侧找最小值替换当前节点，然后删除右侧最小值节点
            BSTNode<K, V> su = minNode(node.right);
            super.remove(su.key);
            node.key = su.key;
            node.vlaue = su.vlaue;
            reBalance(su.parent);
        }else{
            //如果只有左孩子或有左右孩子，在左侧找最大值替换当前节点，然后删除左侧最大值节点
            BSTNode<K, V> pr = maxNode(node.left);
            super.remove(pr.key);
            node.key = pr.key;
            node.vlaue = pr.vlaue;
            reBalance(pr.parent);
        }
    }

    private void reBalance(BSTNode<K, V> p) {
        if (p == null)    return;

        int leftHeight = getHight(p.left);
        int rightHeight = getHight(p.right);

        if (leftHeight - rightHeight > 1){
            //左侧高
            rightRotate(p, p.left);
            reBalance(p.left);
        }else if (rightHeight - leftHeight > 1){
            //右侧高
            leftRotate(p, p.right);
            reBalance(p.right);
        }else{
            reBalance(p.parent);
        }
    }
}

红黑树

  • 要求不那么严格的平衡树一红黑树
  • (1).节点都有颜色标记,且只能是红色或黑色。
  • (2).根是黑色。
  • (3).所有叶子都是黑色(叶子是NIL/nill_节点,不保存实际的数据)。
  • (4).每个红色节点必须有两个黑色的子节点,也可以表述为从每个叶子到根的所有路径上不能有两个连续的红色节点。
  • (5).从任一节点到其每个叶子的所有简单路径都包含相同数目的黑色节点。
  • 根黑，红黑,叶黑,红不连,同黑

实现
public class RedBlackTree<K, V> extends BinarySearchTree<K, V> implements IBinarySearchTree<K, V> {
    private static final boolean RED = true;
    private static final boolean BLACK = false;

    public RedBlackTree(BSTNode<K, V> root) {
        super(root);
        colorBlack(root);
    }

    @Override
    public BSTNode<K, V> insert(K key, V value) {
        //插入节点
        BSTNode<K, V> node = super.insert(key, value);
        //调整插入节点后的红黑树
        adjust(node);
        //根节点染黑
        colorBlack(root);
        size++;
        return node;
    }

    //插入新节点后重新调整红黑树
    private void adjust(BSTNode<K, V> node){
        if (node == null)     return;
        BSTNode<K, V> parent = node.parent;
        if (parent == null)    return;
        //如果父节点为红色破坏了红不连的规则
        if (parent.isRed == RED){
            BSTNode<K, V> grand = parent.parent;
            BSTNode<K, V> uncle = parent.isLeftChild ? parent.parent.right : parent.parent.left;

            if (uncle != null && uncle.isRed == RED){
                //如果uncle为红，gpu反色
                colorRed(grand);
                colorBlack(parent);
                colorBlack(uncle);
                adjust(grand);
            }else{
                //uncle为黑的情况
                if (node.isLeftChild && parent.isLeftChild){
                    //左左型 pg反色 向右旋转
                    colorBlack(parent);
                    colorRed(grand);
                    rightRotate(grand, parent);
                }else if (parent.isLeftChild && !node.isLeftChild){
                    //左右型 沿着p旋转 变成左左型
                    leftRotate(parent, node);
                    adjust(parent);
                }else if (!parent.isLeftChild && node.isLeftChild){
                    //右左型 沿着p旋转 变成右右型
                    rightRotate(parent, node);
                    adjust(parent);
                }else{
                    //右右型 pg反色 向左旋转
                    colorBlack(parent);
                    colorRed(grand);
                    leftRotate(grand, parent);
                }
            }
        }
    }

    private void colorRed(BSTNode<K, V> node){
        if (node != null)    node.isRed = RED;
    }

    private void colorBlack(BSTNode<K, V> node){
        if (node != null)    node.isRed = BLACK;
    }

    @Override
    public void remove(K key) {
        BSTNode<K, V> del = lookupNode(key);
        if (del == null)    return;

        //如果有左右孩子 转移为单支或叶子节点
        if (del.left != null && del.right != null){
            //从右子树中找到最小值
            BSTNode<K, V> n = lookupNode(successor(key));
            del.key = n.key;
            del.vlaue = n.vlaue;
            //del指向最小值节点 为待删除节点
            del = n;
        }

        //此时待删除节点一定为单支或者叶子节点
        //找到顶替节点
        BSTNode<K, V> n = null;
        if (del.left != null){
            n = del.left;
        }else{
            n = del.right;
        }

        if (n == null){    //待删除节点为叶子
            if (del.parent == null){
                //待删除节点为根节点
                root = null;
                return;
            }
            //如果删除的叶子节点为黑色需要先修复
            if (del.isRed == BLACK){
                repair(del);
            }
            //删除叶子
            if (del.parent.isLeftChild){
                del.parent.left = null;
            }else{
                del.parent.right = null;
            }
            del.parent = null;
            del = null;
        }else{
            //单支 使用孩子节点顶替待删除节点
            n.parent = del.parent;
            if (n.parent == null){
                root = n;
                colorBlack(root);
            }
            n.isLeftChild = del.isLeftChild;
            if (n.isLeftChild){
                n.parent.left = n;
            }else{
                n.parent.right = n;
            }
            del.left = del.right = del.parent = null;

            if (del.isRed == BLACK){
                repair(n);
            }
        }
        size--;
    }

    //删除后修复红黑树
    private void repair(BSTNode<K, V> n) {
        if (n.isRed == RED){
            //顶替节点为红色直接染黑
            colorBlack(n);
        }else if (n.parent != null){
            //n为黑 double black 有若干种情况
            case2(n);
        }
    }

    //sib is red 父兄反色     旋转
    private void case2(BSTNode<K, V> n) {
        BSTNode<K, V> sib = getSib(n);
        if (sib.isRed){
            colorBlack(sib);
            colorRed(n.parent);
            if (n == n.parent.left){
                leftRotate(n.parent, sib);
            }else{
                rightRotate(n.parent, sib);
            }
        }
        case3(n);
    }

    //parent is black 且sib双子为黑  sib->red 递归p->case1
    private void case3(BSTNode<K, V> n) {
        BSTNode<K, V> sib = getSib(n);
        boolean sibL = sib.left == null ? BLACK : sib.left.isRed;
        boolean sibR = sib.right == null ? BLACK : sib.right.isRed;
        if (n.parent.isRed == BLACK && sibL == BLACK && sibR == BLACK){
            colorRed(sib);
            repair(n.parent);
        }else{
            case4(n);
        }
    }

    //双子为黑 父亲为红 父兄反色
    private void case4(BSTNode<K, V> n) {
        BSTNode<K, V> sib = getSib(n);
        boolean sibL = sib.left == null ? true : sib.left.isRed;
        boolean sibR = sib.right == null ? true : sib.right.isRed;
        if (n.parent.isRed == RED && sibL == BLACK && sibR == BLACK){
            colorRed(sib);
            colorBlack(n.parent);
        }else{
            case5(n);
        }
    }

    //兄弟向内的孩子为红，通过旋转转移为case6：向外的孩子为红
    private void case5(BSTNode<K, V> n) {
        BSTNode<K, V> sib = getSib(n);
        boolean sibL = sib.left == null ? true : sib.left.isRed;
        boolean sibR = sib.right == null ? true : sib.right.isRed;
        if (n.isLeftChild && sibR == BLACK){
            colorBlack(sib.left);
            colorRed(sib);
            rightRotate(sib, sib.left);
        }else if (!n.isLeftChild && sibL == BLACK){
            colorBlack(sib.right);
            colorRed(sib);
            rightRotate(sib, sib.right);
        }
        case6(n);
    }

    //兄弟向外的孩子为红 兄弟染为父节点的颜色 父节点染黑 父节点旋转
    private void case6(BSTNode<K, V> n) {
        BSTNode<K, V> sib = getSib(n);
        sib.isRed = n.parent.isRed;
        colorBlack(n.parent);
        if (n.isLeftChild){
            colorBlack(sib.right);
            leftRotate(n.parent, sib);
        }else{
            colorBlack(sib.left);
            rightRotate(n.parent, sib);
        }
    }

    private BSTNode<K, V> getSib(BSTNode<K, V> n){
        if (n.isLeftChild){
            return n.parent.right;
        }else{
            return n.parent.left;
        }
    }

    @Override
    public String toString() {
        StringBuilder sb = new StringBuilder();
        List<List<BSTNode<K, V>>> res = levelOrder();
        for (List<BSTNode<K, V>> list : res) {
            for (BSTNode<K, V> bstNode : list) {
                sb.append(bstNode + "\t");
            }
            sb.append("\n");
        }
        return sb.toString();
    }
}

      //二叉树第k层节点数目
public int levelKNum(BSTNode<K, V> root, int k){
  //得到层序遍历下第k层的编号范围
  int min = (int) Math.pow(2, k - 1);
  int max = (int) Math.pow(2, k) - 1;

  //初始化队列辅助宽搜
  Queue<BSTNode<K, V>> qu = new LinkedList<>();
  root.num = 1;
  qu.add(root);
  int count = 0;
  while (!qu.isEmpty()){
      BSTNode<K, V> t = qu.poll();
      if (t.left != null){
          t.left.num = t.num * 2;
          qu.add(t.left);
      }
      if (t.right != null){
          t.right.num = t.num * 2 + 1;
          qu.add(t.right);
      }
      //判断节点编号是否在k层编号范围内
      if (t.num >= min && t.num <= max){
          count++;
      }
      if (t.num > max){
          break;
      }
  }

  return count;
}

前缀树

  • Trie;又称单词查找树;是一-种树形结构;用于保存大量的字符串。它的优点是:利用字符串的公共前缀来节约存储空间
  • Trie 树主要是利用词的公共前缀缩小查词范围、通过状态间的映射关系避免了字符的遍历,从而达到高效检索的目的。

public class Trie {
    TrieNode root;

    public Trie(){
        root = new TrieNode();
    }
    //插入单词
    public void insert(String str){
        char[] chars = str.toCharArray();
        TrieNode p = root;
        for (int i = 0; i < chars.length; i++) {
            TrieNode t = p.children[chars[i] - 'a'];
            if (t == null){
                t = new TrieNode();
                t.level = chars[i] - 'a';
                t.parent = p;
                p.children[chars[i] - 'a'] = t;
            }else{
                t.fre++;
            }
                p = t;
        }

        p.isLast = true;
    }
    //查找前缀为prefix的单词
    public void search(String prefix){
        char[] chars = prefix.toCharArray();
        TrieNode p = root;
        for (int i = 0; i < chars.length; i++) {
            TrieNode t = p.children[chars[i] - 'a'];
            if (t == null){
                return;
            }else{
                p = t;
            }
        }

        print("", p);
    }
    //打印以p为起点的所有单词
    private void print(String prefix, TrieNode p) {
        if (p.isLast && prefix.length() > 0){
            System.out.println(prefix + " " + p.fre);
        }
        for (int i = 0; i < 26; i++) {
            if (p.children[i] != null){
                print(prefix + (char)('a' + p.children[i].level), p.children[i]);
            }
        }
    }
    //打印整个前缀树中所有单词
    public void printAll(){
        print("", root);
    }
}

}

public static TreeNode lowestCommonAncestor(TreeNode root, TreeNode p, TreeNode q) {
    if (root == null || p == null || q == null){
        return null;
    }
    //如果root和p或q相同则直接返回
    if (root.val == p.val || root.val == q.val){
        return root;
    }
    //判断p和q是否在该节点的左和右
    boolean left = f(root.left, p.val);
    boolean right = f(root.right, q.val);
    //如果分别在左右或分别不在左右 则证明p和q分属该节点两侧
    //只要分属某个节点两侧则该节点一定为首个共同祖先
    if (left == right){
        return root;
    }
    //如果都存在于左侧则递归去左侧找
    if (left){
        return lowestCommonAncestor(root.left, p, q);
    }
    //如果都存在于右侧则递归去右侧找
    if (right){
        return lowestCommonAncestor(root.right, p, q);
    }

    return null;
}

        return root;
    }
    //x 为lca || p || q || null 代表找到了lca或者p在这一侧或者q在这一侧或者什么都不是
    TreeNode x = lowestCommonAncestor(root.left, p, q);
    if (x != null && x.val != p.val && x.val != q.val){
        return x;
    }
    TreeNode y = lowestCommonAncestor(root.right, p, q);
    if (y != null && y.val != p.val && y.val != q.val){
        return y;
    }

    if (x != null && y != null){
        return root;
    }else if (root.val == p.val || root.val == q.val){
        return root;
    }else{
        return x != null ? x : y;
    }
}
  1. 检查子树
public static boolean checkSubTree(TreeNode t1, TreeNode t2) {
    if (t1 == null || t2 == null)    return false;
    //找到t1子树中和t2根节点相同的节点
    TreeNode node = s(t1, t2);
    //如果找不到t2一定不是子树
    if (node == null)    return false;

    //判断两个子树是否完全相同
    if (f(node, t2)){
        return true;
    }else{
        //如果不相同则递归检查左侧和右侧
        return checkSubTree(t1.left, t2) || checkSubTree(t1.right, t2);
    }
}
//判断两个子树是否相同
public static boolean f(TreeNode t1, TreeNode t2){
    if (t1 == null && t2 == null)    return true;
    if (t1 == null || t2 == null)    return false;

    if (t1.val != t2.val)    return false;

    return f(t1.left, t2.left) && f(t1.right, t2.right);
}

//在t1子树中找到和n相同的节点
public static TreeNode s(TreeNode t1, TreeNode n){
    if (t1 == null)    return null;
    if (t1.val == n.val){
        return t1;
    }
    TreeNode node = s(t1.left, n);
    if (node != null){
        return node;
    }else{
        return s(t1.right, n);
    }
}

//结果集
static List<List<Integer>> list = new ArrayList<>();
public static void f(List<Integer> cur, TreeNode n, int target){
    if (n == null)    return;
    //l中记录从根节点到叶子的路径
    List<Integer> l = new ArrayList<>();
    l.addAll(cur);
    l.add(n.val);
    //如果该节点为叶子节点则结算
    if (n.left == null && n.right == null){
        //如果路径和正好相等
        if (target - n.val == 0){
            //加入结果集
            list.add(l);
        }
        return;
    }
    f(l, n.left, target - n.val);
    f(l, n.right, target - n.val);
}

//从任意节点向下到任意节点的路径和
private static void help(TreeNode root, int sum) {
    if (root == null)    return;
    f2(new ArrayList<>(), root, sum);
    help(root.left, sum);
    help(root.right, sum);
}

//从根节点到任意向下节点的路径和
public static void f2(List<Integer> cur, TreeNode n, int target){
    if (n == null)    return;
    //l中记录从根节点到叶子的路径
    List<Integer> l = new ArrayList<>();
    l.addAll(cur);
    l.add(n.val);
    //如果该节点为叶子节点或路径和正好相等则结算
    if (target - n.val == 0){
        //加入结果集
        list.add(l);
    }
    if (n.left == null && n.right == null){
        return;
    }
    f2(l, n.left, target - n.val);
    f2(l, n.right, target - n.val);
}
