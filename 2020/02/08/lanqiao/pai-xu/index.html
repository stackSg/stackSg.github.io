分治法

  • 分治法(divide and conquer，D&C) :将原问题划分成若干个规模较小而结构与原问题一致的子问题;递归地解决这些子问题,然后再合并其结果,就得到原问题的解。
  • 容易确定运行时间,是分治算法的优点之一。
  • 分治模式在每一层递归 上都有三个步骤
    -分解(Divide) :将原问题分解成一系列子问题;
    -解决(Conquer):递归地解各子问题。若子问题足够小,则直接有解;
    -合并(Combine):將子问题的结果合并成原问题的解。

//交换数组中的两个元素
private static void swap(int[] arr, int a, int b){
    int t = arr[a];
    arr[a] = arr[b];
    arr[b] = t;
}

  • 快速排序的划分方法：

    1. 单向扫描法

      //单向扫描法
private static int partition(int[] arr, int p, int r) {
 int pivot = arr[p];    //选取第一个元素为key值
 int sp = p + 1;    //扫描指针 从左往右扫描
 int bigger = r;    //bigger指针 右侧为大于key值得元素
 //sp向右扫描直到和bigger交错
 while (sp <= bigger){
     //sp一直向右直到找到大于p的元素
     if (arr[sp] <= pivot){
         sp++;
     }else{
         //把大于p的元素交换到最右侧
         swap(arr, sp, bigger);
         bigger--;    //bigger指针左移，增加一个大于p的元素
     }
 }
 //将选取的key值交换到中间位置
 swap(arr, p, bigger);

 return bigger;
}

    2. 双向扫描法

      //双向扫描法
private static int partition2(int[] arr, int p, int r){
 int pivot = arr[p];    //把第一个元素作为key值
 int left = p + 1;    //左指针
 int right = r;        //右指针

 while (left <= right){
     //left向右扫描找到第一个大于key的元素
     while (left <= right && arr[left] <= pivot){
         left++;
     }
     //right向左扫描找到第一个小于key的元素
     while (left <= right && arr[right] > pivot){
         right--;
     }
     //如果left和right没有交错
     if (left < right){
         //交换两个元素
         swap(arr, left, right);
     }
 }
 //把key值交换到中间
 swap(arr, p, right);
 return right;
}

    3. 有相同元素值的快速排序–三分法

      • 增加e指针，将相同元素放到中间区域
        //三向切分法
private static void sort(int[] arr, int lo, int hi){
if (hi <= lo)    return;
int lt = lo, i = lo + 1, gt = hi;
int key = arr[lo];
while (i <= gt){
  if (arr[i] < key){
      swap(arr, i++, lt++);
  }else if (arr[i] > key){
      swap(arr, i, gt--);
  }else{
      i++;
  }
}
sort(arr, lo, lt - 1);
sort(arr, gt + 1, hi);
}

//归并排序
public static void mergeSort(int[] arr, int l, int r){
    if (l < r){
        //得到中间下标
        int mid = l + ((r-l)>>1);
        //左侧排序
        mergeSort(arr, l, mid);
        //右侧排序
        mergeSort(arr, mid+1,r);
        //合并
        merge(arr, l, mid, r);
    }
}
//将数组中有序的两段[l,mid],[mid+1,r]合并为一段
private static void merge(int[] arr, int l, int mid, int r) {
    int[] arrCopy = Arrays.copyOf(arr, arr.length);    //拷贝原数组
    int left = l;    //左指针指向左边区域开始
    int right = mid+1;    //右指针指向右边开始
    int p = l;    //p指向原数组开始位置

    //合并两段有序元素
    while (left <= mid && right <= r){
        if (arrCopy[left] <= arrCopy[right]){
            arr[p++] = arrCopy[left++];
        }else{
            arr[p++] = arrCopy[right++];
        }
    }
    //左边区域没有拷贝完成的情况继续拷贝 右边区域出现此情况不需要拷贝
    while (left <= mid){
        arr[p++] = arrCopy[left++];
    }
}

  2. 第k个元素

    • 以尽量高的效率求出一个乱序数组中按数值顺序的第k个元素值
      //第k个元素
public static int selectK(int[] arr, int l, int r, int k){
  //分区
  int p = QuickSort.partition2(arr, l, r);
  int q = p;
  //判断分区下标是否为查找的第k个元素
  if (k == q){
      return arr[q];
  }else if (k < q){
      //在左侧继续查找
      return selectK(arr, l, p - 1, k);
  }else{
      //在右侧继续查找
      return selectK(arr, p+1, r, k);
  }
}

  3. 超过一半的数字

    • 数组中有一个数字出现的次数超过了数组长度的一半，找出这个数字。

    • 排序后数组的第length/2个元素一定为所求数字。

      //超过数组一半的数字
public static void fun(){
  int[] arr = {1, 2, 2, 2, 3, 2, 2, 4, 5};
  //利用查找第k个元素，最中间的元素一定为超过数组一半的元素
  System.out.println(selectK(arr, 0, arr.length - 1, arr.length / 2));
  //消除法
  int num = arr[0];
  int n = 1;
  for (int i = 1; i < arr.length; i++) {
      //如果个数消到0则换下一个元素
      if (n == 0){
          num = arr[i];
          n = 1;
      }
      //如果相同则累加，不同则消除
      if (num == arr[i]){
          n++;
      }else{
          n--;
      }
  }

  System.out.println(num);
}

public static void fun2(){
  int[] arr = {1,1,2,3,1,4,5,1};

  int num = arr[0];    //存储数组元素
  int n = 0;    //元素出现的次数
  int endCount = 0;    //最后一个元素出现的次数

  for (int i = 0; i < arr.length; i++) {
      //统计最后一个元素出现次数
      if (arr[i] == arr[arr.length - 1]){
          endCount++;
      }
      //如果两两元素消除则继续下一元素
      if (n == 0){
          num = arr[i];
          n = 1;
          continue;
      }
      //相同累加 不同消减
      if (arr[i] == num){
          n++;
      }else{
          n--;
      }
  }
  //正好等于一半的元素一定在最后一个或者消减完之后Num中存储的值
  if (endCount == arr.length / 2){
      System.out.println(arr[arr.length - 1]);
  }else{
      System.out.println(num);
  }
}

public static int minId2(int[] arr){
  int[] help = new int[arr.length+1];

  int i = 0;
  for (i = 0; i < arr.length; i++) {
      //防止数组中出现大于下标的数字
      if (arr[i] < help.length)
      help[arr[i]] = 1;

      if (help[i] == 0) return i;
  }

//最小可用id-->3.分区法
public static int minId3(int[] arr, int l, int r){
  //l和r交错后结果为l+1
  if (l > r) return l + 1;
  //分区
  int mid = QuickSort.partition2(arr, l, r);
  //t为第几个元素
  int t = mid + 1;
  //如果元素值等于它所在的位置个数则证明左侧连续，需要去右侧找
  if (arr[mid] == t){
      return minId3(arr, mid + 1, r);
  }else{
      return minId3(arr, l, mid - 1);
  }
}

堆排序

  • 堆（英语：heap)是计算机科学中一类特殊的数据结构的统称。堆通常是一个可以被看做一棵树的数组对象。堆总是满足下列性质：
  1. 堆中某个节点的值总是不大于或不小于其父节点的值；
  2. 堆总是一棵完全二叉树。
// 小顶堆调整
public static void minHeapFixDown(int[] arr, int index, int n) {
    // 找到左右孩子
    int left = index * 2 + 1;
    int right = index * 2 + 2;
    // 如果是叶子节点则返回
    if (left >= n && right >= n) {
        return;
    }
    // 找到左右孩子中较小的那个
    int min = left;
    if (right >= n) {
        min = left; // 如果只有左子树则最小值只能为左子树
    } else {
        if (arr[right] < arr[left]) {
            min = right;
        }
    }
    // 如果根节点小于两个孩子节点则不需要调整
    if (arr[index] < arr[min]) {
        return;
    }

    // 把较小的元素调整到父节点
    int temp = arr[index];
    arr[index] = arr[min];
    arr[min] = temp;

    // 对哪个子树进行了调整则递归调整该子树
    minHeapFixDown(arr, min, n);
}

// 大顶堆调整
public static void maxHeapFixDown(int[] arr, int index, int n) {
    // 找到左右孩子
    int left = index * 2 + 1;
    int right = index * 2 + 2;
    // 如果是叶子节点则返回
    if (left >= n && right >= n) {
        return;
    }
    // 找到左右孩子中较大的那个
    int max = left;
    if (right >= n) {
        max = left; // 如果只有左子树则最小值只能为左子树
    } else {
        if (arr[right] > arr[left]) {
            max = right;
        }
    }
    // 如果根节点小于两个孩子节点则不需要调整
    if (arr[index] > arr[max]) {
        return;
    }

    // 把较大的元素调整到父节点
    int temp = arr[index];
    arr[index] = arr[max];
    arr[max] = temp;

    // 对哪个子树进行了调整则递归调整该子树
    maxHeapFixDown(arr, max, n);
}

桶排序

  • 时间复杂度：O(N)~O(NlogN)

  • 实现：

    public static void bucketSort(int[] arr){
  //得到数组最大值
  int max = CountSort.maxOfArray(arr);
  //申明桶空间
  LinkedNode[] bucket = new LinkedNode[arr.length];
  for (int i = 0; i < arr.length; i++) {
      //根据hash函数得到桶下标
      int index = hash(arr[i], max, arr.length);
      //放入桶中
      insert(bucket, arr[i], index);
  }

  //从桶中读出
  int i = 0;
  for (LinkedNode node : bucket) {
      LinkedNode p = node;
      while (p != null){
          arr[i++] = p.data;
          p = p.next;
      }
  }
}
private static void insert(LinkedNode[] bucket, int data, int index) {
  LinkedNode node = new LinkedNode();
  node.data = data;
  if (bucket[index] == null){
      //如果这个桶为空则之间赋值元素
      bucket[index] = node;
  }else{
      //不为空则找按从小到大找到存放位置
      LinkedNode p = bucket[index];
      LinkedNode pre = p;
      while (p != null && data > p.data){
          pre = p;
          p = p.next;
      }

      node.next = pre.next;
      pre.next = node;
  }
}
//hash函数
private static int hash(int i, int max, int length) {
  return (i * length) / (max + 1);
}

        //依据每一位的大小放入桶中排序
        for (int i = 1; i <= num; i++){
            sort(arr, i);
        }

        System.out.println(Arrays.toString(arr));
    }

    private static void sort(int[] arr, int i) {
        //根据第i位的大小放入桶中
        for (int j = 0; j < arr.length; j++) {
            bucket[(int) ((arr[j] / Math.pow(10, i-1))%10)].add(arr[j]);
        }

        //将桶中的数据放入原数组
        int p = 0;
        for (int j = 0; j < bucket.length; j++) {
            for (int k : bucket[j]){
                arr[p++] = k;
            }
        }

        //把桶清空
        clear();
    }

    private static void clear() {
        for (int j = 0; j < bucket.length; j++) {
            bucket[j].clear();
        }
    }

}

      if (sum == n){
          System.out.println("(" + arr[i] + "," + arr[j] + ")");
          i++;
          j--;
      }else if (sum < n){
          i++;
      }else{
          j--;
      }
  }

// 找到无序子数组(升序)
public static int[] findByAsc(int[] arr) {
  int p1 = -1;
  int p2 = -1;
  int max = arr[0];
  // 寻找右边界
  for (int i = 1; i < arr.length; i++) {
      // 找到最大值
      if (arr[i] > max) {
          max = arr[i];
      }
      // 只要是在最大值右侧并且小于最大值的一定为无序
      if (max > arr[i]) {
          p2 = i;
      }
  }
  // 寻找左边界
  int min = arr[arr.length - 1];
  for (int i = arr.length - 2; i >= 0; i--) {
      // 找到历史最小值
      if (arr[i] < min) {
          min = arr[i];
      }
      // 只要是在最小值左侧并且大于最小值的一定无序
      if (arr[i] > min) {
          p1 = i;
      }
  }

  if (p1 == -1) {
      return new int[] { 0, 0 };
  }

  return new int[] { p1, p2 };
}

// 找到无序子数组(降序)
public static int[] findByDesc(int[] arr) {
  int p1 = -1;
  int p2 = -1;
  int min = arr[0];
  // 寻找右边界
  for (int i = 1; i < arr.length; i++) {
      // 找到最小值
      if (arr[i] < min) {
          min = arr[i];
      }
      // 只要是在最小值右侧并且大于最小值的一定为无序
      if (min < arr[i]) {
          p2 = i;
      }
  }
  // 寻找左边界
  int max = arr[arr.length - 1];
  for (int i = arr.length - 2; i >= 0; i--) {
      // 找到历史最大值
      if (arr[i] > max) {
          max = arr[i];
      }
      // 只要是在最大值左侧并且小于最大值的一定无序
      if (arr[i] < max) {
          p1 = i;
      }
  }

  if (p1 == -1) {
      return new int[] { 0, 0 };
  }

  return new int[] { p1, p2 };
}

// 找到无序子数组(升/降序)
public static int[] find(int[] arr) {
  int[] asc = findByAsc(arr);
  int ascL = asc[1] - asc[0] + 1;
  int[] desc = findByDesc(arr);
  int descL = desc[1] - desc[0] + 1;

  if (ascL >= descL) {
      return desc;
  } else {
      return asc;
  }
}

  2. 前k个数

    • 求海量数据(正整数)按逆序排列的前k个数( topK) , 因为数据量太大,不能全部存储在内存中,只能一个一个地从磁盘或者网络
      上读取数据,请设计一个高效的算法来解决这个问题不限制用户输入数据个数,用户每输入-个数据就回车使得程序可立即获得这个数据,用户输入-1代表输入终止，先输入K ,代表要求得topK

    • 请输出topK ,从小到大,空格分割

    • 解：

      // 迭代方法 O(NK)
public static int[] topK() {
  Scanner input = new Scanner(System.in);
  int k = input.nextInt();
  int[] arr = new int[k];
  int n;
  do {
      n = input.nextInt();
      //得到数组中的最小值
      int minInd = min(arr);
      //如果输入的数大于数组最小值则替换此最小值
      if (n > arr[minInd]) {
          arr[minInd] = n;
      }
  } while (n != -1);

  return arr;
}

// 得到数组最小值 O(N)
private static int min(int[] arr) {
  int minInd = 0;
  for (int i = 1; i < arr.length; i++) {
      if (arr[i] < arr[minInd]) {
          minInd = i;
      }
  }

  return minInd;
}

//使用小顶堆 O(NlogK)
static int k;    //前k个数字
static int[] heap;    //堆内存
static int cnt;    //堆中已存在的元素个数
public static int[] topK2() {
  Scanner input = new Scanner(System.in);
  k = input.nextInt();
  heap = new int[k];
  int n;
  do {
      n = input.nextInt();
      fun(n);
  } while (n != -1);

  return heap;
}

private static void fun(int n) {
  //在堆填满之前只需要输入数据
  if (cnt < k - 1){
      heap[cnt++] = n;
      return;
  }

  if (cnt == k - 1){    //当堆填到只剩一个元素时
      heap[cnt++] = n;
      //初始化堆
      BinaryTreeAndArray.heap(heap);
  }else{
      //如果输入数据大于小顶堆堆顶
      if (n > heap[0]){
          //替换堆顶元素
          heap[0] = n;    
          //调整堆
          BinaryTreeAndArray.minHeapFixDown(heap, 0, heap.length);    
      }
  }
  System.out.println(Arrays.toString(heap));
}

  3. 所有员工年龄排序

    • 公司现在要对几万员工的年龄进行排序,因为公司员工的人数非常多,所以要求排序算法的效率要非常高,你能写出这样的程序吗
      输入:输入可能包含多个测试样例,对于每个测试案例,输入的第一-行为- 一个整数n(1<= n<=1000000) :代表公司内员工的人数。
      输入的第二行包括n个整数:代表公司内每个员工的年龄。其中，员工年龄
      age的取值范围为(1<=age<=99)。

    • 输出:对应每个测试案例,请输出排序后的n个员工的年龄，每个年龄后面有一个空格。

    • 解：使用计数排序

      public static void main(String[] args) {
  //得到员工年龄
  int ages[] = new int[10000];
  Random r = new Random();
  for (int i = 0; i < ages.length; i++) {
      ages[i] = r.nextInt(100);
  }
  System.out.println(Arrays.toString(ages));
  //放入计数桶中
  int[] bucket = new int[100];
  for (int i = 0; i < ages.length; i++) {
      bucket[ages[i]]++;
  }

  //得到排序结果
  int p = 0;
  for (int i = 0; i < bucket.length; i++) {
      while (bucket[i] != 0){
          ages[p++] = i;
          bucket[i]--;
      }
  }
  System.out.println(Arrays.toString(ages));
}

  5. 字符串(数组)的包含

public static boolean fun(String s1, String s2){
  char[] s1Arr = s1.toCharArray();
  //排序
  Arrays.sort(s1Arr);

      }
  }

  return true;
}
