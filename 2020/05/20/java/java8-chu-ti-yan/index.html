Lambda表达式

@FunctionalInterface
public interface MyFilter {
    String getValue(String str);
}

public interface MyFun<T> {
    T getValue(T t1, T t2);
}

@Test
public void lambda(){
    int i = 0;
    Runnable runnable = ()-> System.out.println(i);
    runnable.run();
    //如果参数只有一个，()可以省略
    Consumer<String> consumer = x-> System.out.println(x);
    consumer.accept("abc");

    Comparator<Integer> comparator = (x, y)->{
        System.out.println("111");
        return Integer.compare(x, y);
    };
    //如果只有一条语句{}可以省略，return 可以省略
    Comparator<Integer> comparator1 = (x, y)->Integer.compare(x,y );
    System.out.println(comparator1.compare(1,2));

    List<String> list = Arrays.asList(
            "zhangsan",
            "lisi",
            "wangwu"
    );
    Collections.sort(list, (s1, s2)->-s1.compareTo(s2));
    list.forEach(System.out::println);

    System.out.println(handler("abc", (s)->s.toUpperCase()));

    System.out.println(handler(100l, 200l, (x, y)->x*y));
    System.out.println(handler(10l, 20l, (x, y)->x+y));
}

public String handler(String str, MyFilter filter){
    return filter.getValue(str);
}

public Long handler(Long l1, Long l2, MyFun<Long> myFun){
    return myFun.getValue(l1, l2);
}

函数式接口

java8内置的四大核心函数式接口

  1. Consumer : 消费型接口
    void accept(T t);
  2. Supplier : 供给型接口
    T get();
  3. Function<T, R> : 函数型接口
    R apply(T t);
  4. Predicate : 断言型接口
    boolean test(T t);
/*消费型接口*/
public void money(int money, Consumer<Integer> consumer){
    consumer.accept(money);
}

/*供给型接口*/
public List<Integer> getNumList(int count, Supplier<Integer> supplier){
    List<Integer> list = new ArrayList<>();
    for (int i = 0; i < count; i++) {
        list.add(supplier.get());
    }
    return list;
}

/*函数型接口*/
public String strHandler(String str, Function<String, String> function){
    return function.apply(str);
}

/*断言型接口*/
public List<String> filterStr(List<String> list, Predicate<String> predicate){
    List<String> res = new ArrayList<>();
    for (String s : list){
        if (predicate.test(s)){
            res.add(s);
        }
    }
    return res;
}

方法引用和构造器引用

  1. 方法引用语法格式：
    对象：：实例方法名
    类：：静态方法名
    类：：实例方法名
  2. 构造器引用：
    classname::new
  3. 数组引用：
    type::new

Stream API

  • “集合讲的是数据，流讲的是计算”
class User implements Comparable<User>{
    private int id;
    private String name;
    private int age;

    public User() {
    }

    public User(String name, int age) {
        this.name = name;
        this.age = age;
    }

    public User(int id, String name, int age) {
        this.id = id;
        this.name = name;
        this.age = age;
    }

    public int getId() {
        return id;
    }

    public void setId(int id) {
        this.id = id;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        User user = (User) o;
        return id == user.id &&
                age == user.age &&
                Objects.equals(name, user.name);
    }

    @Override
    public int hashCode() {
        return Objects.hash(id, name, age);
    }

    @Override
    public String toString() {
        return "User{" +
                "id=" + id +
                ", name='" + name + '\'' +
                ", age=" + age +
                '}';
    }

    @Override
    public int compareTo(User o) {
        return this.name.compareTo(o.name);
    }
}

public static Stream<Character> getCharacterStream(String s){
    List<Character> list = new ArrayList<>();
    for (int i = 0; i < s.length(); i++){
        list.add(s.charAt(i));
    }
    return list.stream();
}

@Test
public void searchAndMatch(){
    //全部匹配
    boolean b = users.stream().allMatch((user -> user.getAge() == 30));
    System.out.println(b);
    //有匹配的
    boolean b2 = users.stream().anyMatch((user -> user.getName() == "张三"));
    System.out.println(b2);
    //没有匹配的
    boolean b3 = users.stream().noneMatch((user -> user.getId() == 1));
    System.out.println(b3);
    //返回第一个元素
    Optional<User> first = users.stream().sorted().findFirst();
    System.out.println(first.get());
    //返回任意一个元素
    Optional<User> any = users.stream().filter(user -> user.getAge() == 60).findAny();
    System.out.println(any.get());
    //返回最大值
    Optional<Integer> max = users.stream().map(User::getAge).max(Integer::compareTo);
    System.out.println(max.get());
    //返回最小值
    Optional<Integer> min = users.stream().map(User::getId).min(Integer::compareTo);
    System.out.println(min.get());
}

/*收集*/
@Test
public void collect(){
    //收集到list中
    List<String> list = users.stream().map(User::getName).collect(Collectors.toList());
    System.out.println(list);
    //收集到自定义数据结构中
    HashSet<String> hashSet = users.stream().map(User::getName).collect(Collectors.toCollection(HashSet::new));
    System.out.println(hashSet);
    //统计总数
    Long collect = users.stream().collect(Collectors.counting());
    System.out.println(collect);
    //平均值
    Double age = users.stream().collect(Collectors.averagingInt(User::getAge));
    System.out.println(age);
    //总和
    Integer sum = users.stream().collect(Collectors.summingInt(User::getAge));
    System.out.println(sum);
    //最小值
    Optional<Integer> min = users.stream().map(User::getAge).collect(Collectors.minBy(Integer::compare));
    System.out.println(min.get());
    //最大值
    Optional<Integer> max = users.stream().map(User::getId).collect(Collectors.maxBy(Integer::compareTo));
    System.out.println(max.get());
    //分组
    Map<Integer, List<User>> map = users.stream().collect(Collectors.groupingBy(User::getAge));
    System.out.println(map);
    //多级分组
    Map<String, Map<Integer, List<User>>> map1 = users.stream().collect(Collectors.groupingBy(User::getName, Collectors.groupingBy(User::getAge)));
    System.out.println(map1);
    //分区
    Map<Boolean, List<User>> collect1 = users.stream().collect(Collectors.partitioningBy(user -> user.getId() > 3));
    System.out.println(collect1);
    //统计
    IntSummaryStatistics summaryStatistics = users.stream().collect(Collectors.summarizingInt(User::getAge));
    System.out.println(summaryStatistics.getMax());
    System.out.println(summaryStatistics.getAverage());
    System.out.println(summaryStatistics.getMin());
    System.out.println(summaryStatistics.getCount());
    //连接
    String s = users.stream().map(User::getName).collect(Collectors.joining(",", "*", "$"));
    System.out.println(s);
}
